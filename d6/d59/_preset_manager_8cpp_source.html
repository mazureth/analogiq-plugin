139juce::String PresetManager::serializeRackToJSON(Rack *rack) const
140{
141 if (rack == nullptr)
142 return "{}";
143
144 juce::DynamicObject::Ptr jsonObj = new juce::DynamicObject();
145
146 // Add metadata
147 jsonObj->setProperty("version", "1.0");
148 jsonObj->setProperty("timestamp", juce::Time::getCurrentTime().toMilliseconds());
149
150 // Serialize slots
151 juce::Array<juce::var> slotsArray;
152
153 for (int i = 0; i < rack->getNumSlots(); ++i)
154 {
155 if (auto *slot = rack->getSlot(i))
156 {
157 if (auto *item = slot->getGearItem())
158 {
159 // Only save slots that have gear items
160 juce::DynamicObject::Ptr slotObj = new juce::DynamicObject();
161 slotObj->setProperty("slotIndex", i);
162 slotObj->setProperty("unitId", item->unitId);
163
164 // Save instance information
165 if (item->isInstance)
166 {
167 slotObj->setProperty("instanceId", item->instanceId);
168 slotObj->setProperty("sourceUnitId", item->sourceUnitId);
169 }
170
171 // Serialize controls
172 juce::Array<juce::var> controlsArray;
173 for (int j = 0; j < item->controls.size(); ++j)
174 {
175 const auto &control = item->controls[j];
176 juce::DynamicObject::Ptr controlObj = new juce::DynamicObject();
177 controlObj->setProperty("index", j);
178 controlObj->setProperty("value", control.value);
179 controlObj->setProperty("initialValue", control.initialValue);
180
181 if (control.type == GearControl::Type::Switch || control.type == GearControl::Type::Button)
182 {
183 controlObj->setProperty("currentIndex", control.currentIndex);
184 }
185
186 controlsArray.add(juce::var(controlObj));
187 }
188 slotObj->setProperty("controls", controlsArray);
189
190 slotsArray.add(juce::var(slotObj));
191 }
192 }
193 }
194
195 jsonObj->setProperty("slots", slotsArray);
196
197 return juce::JSON::toString(juce::var(jsonObj));
198}

208bool PresetManager::deserializeJSONToRack(const juce::String &jsonData, Rack *rack, GearLibrary *gearLibrary) const
209{
210 if (rack == nullptr || gearLibrary == nullptr)
211 return false;
212
213 // Parse JSON
214 auto jsonVar = juce::JSON::parse(jsonData);
215 if (!jsonVar.isObject())
216 return false;
217
218 auto jsonObj = jsonVar.getDynamicObject();
219 if (jsonObj == nullptr)
220 return false;
221
222 // Clear current rack
223 for (int i = 0; i < rack->getNumSlots(); ++i)
224 {
225 if (auto *slot = rack->getSlot(i))
226 {
227 slot->clearGearItem();
228 }
229 }
230
231 // Load slots
232 auto slotsVar = jsonObj->getProperty("slots");
233 if (slotsVar.isArray())
234 {
235 auto slotsArray = slotsVar.getArray();
236 for (auto slotVar : *slotsArray)
237 {
238 if (slotVar.isObject())
239 {
240 auto slotObj = slotVar.getDynamicObject();
241 if (slotObj != nullptr)
242 {
243 int slotIndex = slotObj->getProperty("slotIndex");
244 juce::String unitId = slotObj->getProperty("unitId");
245
246 if (slotIndex >= 0 && slotIndex < rack->getNumSlots() && !unitId.isEmpty())
247 {
248 // Find the gear item in the library
249 const auto &items = gearLibrary->getItems();
250 const GearItem *sourceItem = nullptr;
251
252 for (int i = 0; i < items.size(); ++i)
253 {
254 const auto &item = items.getReference(i);
255 if (item.unitId == unitId)
256 {
257 sourceItem = &item;
258 break;
259 }
260 }
261
262 if (sourceItem != nullptr)
263 {
264 // Create a new instance of the gear item
265 GearItem *newItem = new GearItem(*sourceItem, INetworkFetcher::getDummy(), fileSystem, cacheManager);
266
267 // Check if this was an instance and restore instance properties
268 auto instanceIdVar = slotObj->getProperty("instanceId");
269 auto sourceUnitIdVar = slotObj->getProperty("sourceUnitId");
270
271 if (instanceIdVar.isString() && sourceUnitIdVar.isString())
272 {
273 newItem->isInstance = true;
274 newItem->instanceId = instanceIdVar.toString();
275 newItem->sourceUnitId = sourceUnitIdVar.toString();
276 }

661bool PresetManager::validatePresetName(const juce::String &name, juce::String &errorMessage) const
662{
663 errorMessage.clear();
664
665 // Check for empty or whitespace-only names
666 if (name.trim().isEmpty())
667 {
668 errorMessage = "Preset name cannot be empty or contain only whitespace.";
669 return false;
670 }
671
672 // Check for invalid characters
673 if (name.containsAnyOf("<>:\"/\\|?*"))
674 {
675 errorMessage = "Preset name contains invalid characters. The following characters are not allowed: < > : \" / \\ | ? *";
676 return false;
677 }
678
679 // Check for reserved names
680 juce::String lowerName = name.toLowerCase();
681 if (lowerName == "con" || lowerName == "prn" || lowerName == "aux" ||
682 lowerName == "nul" || lowerName == "com1" || lowerName == "com2" ||
683 lowerName == "com3" || lowerName == "com4" || lowerName == "com5" ||
684 lowerName == "com6" || lowerName == "com7" || lowerName == "com8" ||
685 lowerName == "com9" || lowerName == "lpt1" || lowerName == "lpt2" ||
686 lowerName == "lpt3" || lowerName == "lpt4" || lowerName == "lpt5" ||
687 lowerName == "lpt6" || lowerName == "lpt7" || lowerName == "lpt8" ||
688 lowerName == "lpt9")
689 {
690 errorMessage = "Preset name is a reserved system name and cannot be used.";
691 return false;
692 }
693
694 // Check for names that start or end with dots or spaces
695 if (name.startsWith(".") || name.endsWith(".") ||
696 name.startsWith(" ") || name.endsWith(" "))
697 {
698 errorMessage = "Preset name cannot start or end with dots or spaces.";
699 return false;
700 }
701
702 // Check for names that are too long (considering filename conversion)
703 if (name.length() > 200)
704 {
705 errorMessage = "Preset name is too long. Maximum length is 200 characters.";
706 return false;
707 }
708
709 return true;
710}
