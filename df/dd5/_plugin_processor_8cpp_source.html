286void AnalogIQProcessor::saveInstanceStateFromRack(Rack *rack, juce::ValueTree &instanceTree)
287{
288 // Save instance data for each slot
289 for (int i = 0; i < rack->getNumSlots(); ++i)
290 {
291 if (auto *slot = rack->getSlot(i))
292 {
293 if (auto *item = slot->getGearItem())
294 {
295
296 // Save state if this is an instance with a valid instance ID
297 if (item->isInstance && !item->instanceId.isEmpty() && !item->unitId.isEmpty())
298 {
299 auto slotTree = instanceTree.getOrCreateChildWithName("slot_" + juce::String(i), nullptr);
300 slotTree.setProperty("instanceId", item->instanceId, nullptr);
301 slotTree.setProperty("sourceUnitId", item->sourceUnitId, nullptr);
302
303 // Save control values
304 auto controlsTree = slotTree.getOrCreateChildWithName("controls", nullptr);
305 for (int j = 0; j < item->controls.size(); ++j)
306 {
307 const auto &control = item->controls[j];
308 auto controlTree = controlsTree.getOrCreateChildWithName("control_" + juce::String(j), nullptr);
309 controlTree.setProperty("value", control.value, nullptr);
310 controlTree.setProperty("initialValue", control.initialValue, nullptr);
311 if (control.type == GearControl::Type::Switch)
312 {
313 controlTree.setProperty("currentIndex", control.currentIndex, nullptr);
314 }
315 }
316 }
317 else
318 {
319 // TODO: Remove this logging and handle this case
320 }
321 }
322 }
323 }

332void AnalogIQProcessor::loadInstanceState(Rack *rack)
333{
334 // Get the instance state tree
335 auto instanceTree = state.state.getChildWithName("instances");
336 if (!instanceTree.isValid())
337 return;
338
339 if (rack != nullptr)
340 {
341 // Load instance data for each slot
342 for (int i = 0; i < rack->getNumSlots(); ++i)
343 {
344 auto slotTree = instanceTree.getChildWithName("slot_" + juce::String(i));
345 if (slotTree.isValid())
346 {
347 // Get the source unit ID from the saved state
348 auto sourceUnitId = slotTree.getProperty("sourceUnitId").toString();
349 if (!sourceUnitId.isEmpty())
350 {
351 // Create a new gear item from the source
352 // TODO: why is this using inline test data instead of whatever is
353 // stored in the instanceTree?
354 GearItem *item = new GearItem(
355 sourceUnitId, // unitId
356 "Test EQ", // name
357 "Test Co", // manufacturer
358 "equalizer", // categoryString
359 "1.0", // version
360 "", // schemaPath
361 "", // thumbnailImage
362 juce::StringArray(), // tags
363 networkFetcher, // networkFetcher (required)
364 GearType::Series500, // type
365 GearCategory::EQ, // category
366 1, // slotSize
367 juce::Array<GearControl>() // controls
368 );
369
370 // Add controls to match the saved state
371 auto controlsTree = slotTree.getChildWithName("controls");
372 if (controlsTree.isValid())
373 {
374 for (int j = 0; j < controlsTree.getNumChildren(); ++j)
375 {
376 auto controlTree = controlsTree.getChild(j);
377 if (controlTree.isValid())
378 {
379 GearControl control;
380 control.name = "Control " + juce::String(j);
381 control.type = GearControl::Type::Knob;
382 control.value = controlTree.getProperty("value", 0.0f);
383 control.initialValue = controlTree.getProperty("initialValue", 0.0f);
384 item->controls.add(control);
385 }
386 }
387 }
388
389 // Set the gear item in the slot
390 if (auto *slot = rack->getSlot(i))
391 {
392 slot->setGearItem(item);
393
394 // Create instance if we have saved state
395 rack->createInstance(i);
396
397 // Get the gear item for this slot
398 if (auto *item = slot->getGearItem())
399 {
400 // Load control values
401 auto controlsTree = slotTree.getChildWithName("controls");
402 if (controlsTree.isValid())
403 {
404 for (int j = 0; j < item->controls.size(); ++j)
405 {
406 auto controlTree = controlsTree.getChildWithName("control_" + juce::String(j));
407 if (controlTree.isValid())
408 {
409 auto &control = item->controls.getReference(j);
410 control.value = controlTree.getProperty("value", control.value);
411 control.initialValue = controlTree.getProperty("initialValue", control.initialValue);
412 if (control.type == GearControl::Type::Switch)
413 {
414 control.currentIndex = controlTree.getProperty("currentIndex", control.currentIndex);
415 }
416 }
417 }
418 }
419 }
420 else
421 {
422 }
