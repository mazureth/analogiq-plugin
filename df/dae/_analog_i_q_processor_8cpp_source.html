354juce::AudioProcessorEditor *AnalogIQProcessor::createEditor()
355{
356 auto *editor = new AnalogIQEditor(*this, *fileSystem, *cacheManager, *presetManager, *gearLibrary);
357 lastCreatedEditor = editor;
358
359 // Store a reference to the rack for fallback operations when editor is not available
360 if (auto *rackEditor = dynamic_cast<AnalogIQEditor *>(editor))
361 {
362 rack = rackEditor->getRack();
363 logToFile("Stored rack reference for fallback operations");
364
365 // Load instance state after the editor is created and gear library is loaded
366 // We'll defer this to after the gear library is ready
367 juce::MessageManager::callAsync([this, rackEditor]()

369 // Wait a bit more to ensure gear library is fully loaded
370 juce::Timer::callAfterDelay(100, [this, rackEditor]()
371 {
372 if (auto *rack = rackEditor->getRack())
373 {
374 loadInstanceState(rack);
375 }
376 }); });
377 }

534void AnalogIQProcessor::saveInstanceStateFromRack(Rack *rack, juce::ValueTree &instanceTree)
535{
536 logToFile("=== saveInstanceStateFromRack START ===");
537
538 if (rack == nullptr)
539 {
540 logToFile("ERROR: Rack pointer is null");
541 return;
542 }
543
544 logToFile("Rack validation: rack pointer valid");
545 logToFile("Rack slots: " + juce::String(rack->getNumSlots()));
546
547 // Save instance data for each slot
548 for (int i = 0; i < rack->getNumSlots(); ++i)
549 {
550 logToFile("Processing slot " + juce::String(i));
551
552 if (auto *slot = rack->getSlot(i))
553 {
554 logToFile("Slot " + juce::String(i) + " obtained successfully");
555
556 if (auto *item = slot->getGearItem())
557 {
558 logToFile("Slot " + juce::String(i) + " has gear item: " + item->name);
559 logToFile(" Is instance: " + juce::String(item->isInstance ? "true" : "false"));
560 logToFile(" Instance ID: " + item->instanceId);
561 logToFile(" Unit ID: " + item->unitId);
562 logToFile(" Controls count: " + juce::String(item->controls.size()));
563
564 // Save state for instances only (all items in rack are now instances)
565 if (item->isInstance && !item->instanceId.isEmpty() && !item->unitId.isEmpty())
566 {
567 logToFile("Saving instance data for slot " + juce::String(i));
568
569 auto slotTree = instanceTree.getOrCreateChildWithName("slot_" + juce::String(i), &undoManager);
570 logToFile("Slot tree created for slot " + juce::String(i));
571
572 slotTree.setProperty("instanceId", item->instanceId, &undoManager);
573 slotTree.setProperty("sourceUnitId", item->sourceUnitId, &undoManager);
574 logToFile("Slot properties set for slot " + juce::String(i));
575
576 // Save control values
577 auto controlsTree = slotTree.getOrCreateChildWithName("controls", &undoManager);
578 logToFile("Controls tree created for slot " + juce::String(i));
579
580 for (int j = 0; j < item->controls.size(); ++j)
581 {
582 const auto &control = item->controls[j];
583 logToFile("Processing control " + juce::String(j) + " in slot " + juce::String(i));
584 logToFile(" Control name: " + control.name);
585 logToFile(" Control type: " + juce::String(static_cast<int>(control.type)));
586 logToFile(" Control value: " + juce::String(control.value));
587 logToFile(" Control initial value: " + juce::String(control.initialValue));
588
589 auto controlTree = controlsTree.getOrCreateChildWithName("control_" + juce::String(j), &undoManager);
590 controlTree.setProperty("value", control.value, &undoManager);
591 controlTree.setProperty("initialValue", control.initialValue, &undoManager);
592
593 if (control.type == GearControl::Type::Switch || control.type == GearControl::Type::Button)
594 {
595 controlTree.setProperty("currentIndex", control.currentIndex, &undoManager);
596 logToFile(" Control current index: " + juce::String(control.currentIndex));
597 }
598
599 logToFile("Control " + juce::String(j) + " saved successfully");
600 }
601
602 logToFile("Slot " + juce::String(i) + " completed successfully");
603 }
604 else
605 {
606 logToFile("Slot " + juce::String(i) + " skipped - not a valid instance");
607 logToFile(" Is instance: " + juce::String(item->isInstance ? "true" : "false"));
608 logToFile(" Instance ID empty: " + juce::String(item->instanceId.isEmpty() ? "true" : "false"));
609 logToFile(" Unit ID empty: " + juce::String(item->unitId.isEmpty() ? "true" : "false"));
610 }
611 }
612 else
613 {
614 logToFile("Slot " + juce::String(i) + " has no gear item");
615 }
616 }
617 else
618 {
619 logToFile("ERROR: Failed to get slot " + juce::String(i));
620 }
621 }
622
623 // Save notes panel content
624 logToFile("Attempting to save notes panel content...");
625 if (auto *editor = dynamic_cast<AnalogIQEditor *>(getActiveEditor()))
626 {
627 logToFile("Editor available for notes panel");
628 if (auto *notesPanel = editor->getNotesPanel())
629 {
630 logToFile("Notes panel obtained successfully");
631 auto notesTree = instanceTree.getOrCreateChildWithName("notes", &undoManager);
632 auto notesContent = notesPanel->getText();
633 notesTree.setProperty("content", notesContent, &undoManager);
634 logToFile("Notes content saved: " + juce::String(notesContent.length()) + " characters");
635 }
636 else
637 {
638 logToFile("WARNING: Notes panel is null");
639 }
640 }
641 else
642 {
643 logToFile("Editor not available for notes panel");
644 }
645
646 logToFile("Final instance tree children: " + juce::String(instanceTree.getNumChildren()));
647 logToFile("=== saveInstanceStateFromRack END ===");
648}

656void AnalogIQProcessor::loadInstanceState(Rack *rack)
657{
658 // Get the instance state tree
659 auto instanceTree = state.state.getChildWithName("instances");
660 if (!instanceTree.isValid())
661 return;
662
663 if (rack != nullptr)
664 {
665 // Load instance data for each slot
666 for (int i = 0; i < rack->getNumSlots(); ++i)
667 {
668 auto slotTree = instanceTree.getChildWithName("slot_" + juce::String(i));
669 if (slotTree.isValid())
670 {
671 // Get the source unit ID from the saved state
672 auto sourceUnitId = slotTree.getProperty("sourceUnitId").toString();
673 if (!sourceUnitId.isEmpty())
674 {
675 // Load the gear item from the gear library using the source unit ID
676 auto gearItem = gearLibrary->getGearItemByUnitId(sourceUnitId);
677 if (gearItem != nullptr)
678 {
679 // Create a new instance from the source gear using proper copy constructor
680 auto *item = new GearItem(*gearItem, networkFetcher, *fileSystem, *cacheManager);
681
682 // Set the gear item in the slot (this automatically creates an instance)
683 if (auto *slot = rack->getSlot(i))
684 {
685 slot->setGearItem(item);
686
687 // Validate that the instance was created successfully
688 if (item->isInstance && !item->instanceId.isEmpty())
689 {
690 // Get the gear item for this slot
691 if (auto *loadedItem = slot->getGearItem())
692 {
693 // Capture control values before schema loading (following PresetManager pattern)
694 struct SavedControlValues
695 {
696 int index;
697 float value;
698 float initialValue;
699 int currentIndex;
700 };
701 juce::Array<SavedControlValues> savedControls;
702
703 // Extract control values from saved state
704 auto controlsTree = slotTree.getChildWithName("controls");
705 if (controlsTree.isValid())
706 {
707 for (int j = 0; j < controlsTree.getNumChildren(); ++j)
708 {
709 auto controlTree = controlsTree.getChildWithName("control_" + juce::String(j));
710 if (controlTree.isValid())
711 {
712 SavedControlValues saved;
713 saved.index = j;
714 saved.value = controlTree.getProperty("value", 0.0f);
715 saved.initialValue = controlTree.getProperty("initialValue", 0.0f);
716 saved.currentIndex = controlTree.getProperty("currentIndex", 0);
717 savedControls.add(saved);
718 }
719 }
720 }
721
722 // Trigger faceplate and control image loading
723 // We'll load the schema and then restore control values after
724 rack->fetchSchemaForGearItem(loadedItem, [savedControls, loadedItem]()
725 {
726 // Apply saved control values after schema parsing (following PresetManager pattern)
727 // Note: initialValue is preserved from schema, not restored from saved state
728 // Validate loadedItem is still valid
729 if (loadedItem != nullptr && savedControls.size() > 0)
730 {
731 for (const auto& saved : savedControls)
732 {
733 if (saved.index >= 0 && saved.index < loadedItem->controls.size())
734 {
735 auto &control = loadedItem->controls.getReference(saved.index);
736 control.value = saved.value;
737 // Don't restore initialValue - keep the schema default
738
739 if (control.type == GearControl::Type::Switch || control.type == GearControl::Type::Button)
740 {
741 control.currentIndex = saved.currentIndex;
742 }
743 }
744 }
745 } });
746 }
747 }
748 }
749 }
750 }
751 }
752 }
753
754 // Load notes panel content after all gear items are processed
755 if (auto *editor = dynamic_cast<AnalogIQEditor *>(getActiveEditor()))
756 {
757 if (auto *notesPanel = editor->getNotesPanel())
758 {
759 auto notesTree = instanceTree.getChildWithName("notes");
760 if (notesTree.isValid())
761 {
762 auto notesContent = notesTree.getProperty("content").toString();
763 if (notesContent.isNotEmpty())
764 {
765 notesPanel->setText(notesContent);
766 }
767 }
768 }
769 }
770 }
