289void AnalogIQProcessor::saveInstanceStateFromRack(Rack *rack, juce::ValueTree &instanceTree)
290{
291 // Save instance data for each slot
292 for (int i = 0; i < rack->getNumSlots(); ++i)
293 {
294 if (auto *slot = rack->getSlot(i))
295 {
296 if (auto *item = slot->getGearItem())
297 {
298
299 // Save state if this is an instance with a valid instance ID
300 if (item->isInstance && !item->instanceId.isEmpty() && !item->unitId.isEmpty())
301 {
302 auto slotTree = instanceTree.getOrCreateChildWithName("slot_" + juce::String(i), nullptr);
303 slotTree.setProperty("instanceId", item->instanceId, nullptr);
304 slotTree.setProperty("sourceUnitId", item->sourceUnitId, nullptr);
305
306 // Save control values
307 auto controlsTree = slotTree.getOrCreateChildWithName("controls", nullptr);
308 for (int j = 0; j < item->controls.size(); ++j)
309 {
310 const auto &control = item->controls[j];
311 auto controlTree = controlsTree.getOrCreateChildWithName("control_" + juce::String(j), nullptr);
312 controlTree.setProperty("value", control.value, nullptr);
313 controlTree.setProperty("initialValue", control.initialValue, nullptr);
314 if (control.type == GearControl::Type::Switch)
315 {
316 controlTree.setProperty("currentIndex", control.currentIndex, nullptr);
317 }
318 }
319 }
320 else
321 {
322 // TODO: Remove this logging and handle this case
323 }
324 }
325 }
326 }

335void AnalogIQProcessor::loadInstanceState(Rack *rack)
336{
337 // Get the instance state tree
338 auto instanceTree = state.state.getChildWithName("instances");
339 if (!instanceTree.isValid())
340 return;
341
342 if (rack != nullptr)
343 {
344 // Load instance data for each slot
345 for (int i = 0; i < rack->getNumSlots(); ++i)
346 {
347 auto slotTree = instanceTree.getChildWithName("slot_" + juce::String(i));
348 if (slotTree.isValid())
349 {
350 // Get the source unit ID from the saved state
351 auto sourceUnitId = slotTree.getProperty("sourceUnitId").toString();
352 if (!sourceUnitId.isEmpty())
353 {
354 // Create a new gear item from the source
355 // TODO: why is this using inline test data instead of whatever is
356 // stored in the instanceTree?
357 // Use the file system from the cache manager instead of creating a new one
358 GearItem *item = new GearItem(
359 sourceUnitId, // unitId
360 "Test EQ", // name
361 "Test Co", // manufacturer
362 "equalizer", // categoryString
363 "1.0", // version
364 "", // schemaPath
365 "", // thumbnailImage
366 juce::StringArray(), // tags
367 networkFetcher, // networkFetcher (required)
368 *fileSystem, // fileSystem (required)
369 *cacheManager, // cacheManager (required)
370 GearType::Series500, // type
371 GearCategory::EQ, // category
372 1, // slotSize
373 juce::Array<GearControl>() // controls
374 );
375
376 // Add controls to match the saved state
377 auto controlsTree = slotTree.getChildWithName("controls");
378 if (controlsTree.isValid())
379 {
380 for (int j = 0; j < controlsTree.getNumChildren(); ++j)
381 {
382 auto controlTree = controlsTree.getChild(j);
383 if (controlTree.isValid())
384 {
385 GearControl control;
386 control.name = "Control " + juce::String(j);
387 control.type = GearControl::Type::Knob;
388 control.value = controlTree.getProperty("value", 0.0f);
389 control.initialValue = controlTree.getProperty("initialValue", 0.0f);
390 item->controls.add(control);
391 }
392 }
393 }
394
395 // Set the gear item in the slot
396 if (auto *slot = rack->getSlot(i))
397 {
398 slot->setGearItem(item);
399
400 // Create instance if we have saved state
401 rack->createInstance(i);
402
403 // Get the gear item for this slot
404 if (auto *item = slot->getGearItem())
405 {
406 // Load control values
407 auto controlsTree = slotTree.getChildWithName("controls");
408 if (controlsTree.isValid())
409 {
410 for (int j = 0; j < item->controls.size(); ++j)
411 {
412 auto controlTree = controlsTree.getChildWithName("control_" + juce::String(j));
413 if (controlTree.isValid())
414 {
415 auto &control = item->controls.getReference(j);
416 control.value = controlTree.getProperty("value", control.value);
417 control.initialValue = controlTree.getProperty("initialValue", control.initialValue);
418 if (control.type == GearControl::Type::Switch)
419 {
420 control.currentIndex = controlTree.getProperty("currentIndex", control.currentIndex);
421 }
422 }
423 }
424 }
425 }
426 else
427 {
428 }
