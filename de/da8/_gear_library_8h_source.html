182 static juce::String getFullUrl(const juce::String &relativePath)
183 {
184 // If already a full URL, return as is
185 if (relativePath.startsWith("http"))
186 return relativePath;
187
188 // If this is an absolute path on the filesystem, return as is
189 if (relativePath.startsWith("/"))
190 return relativePath;
191
192 // Handle the case where we might need to add assets/ or units/ prefix
193 juce::String result;
194
195 if (relativePath.startsWith("assets/") || relativePath.startsWith("units/"))
196 {
197 // Path already has the correct folder prefix
198 result = RemoteResources::BASE_URL + relativePath;
199 }
200 else if (relativePath.endsWith(".json"))
201 {
202 // Likely a schema file - add units/ prefix if needed
203 result = RemoteResources::BASE_URL + RemoteResources::SCHEMAS_PATH + relativePath;
204 }
205 else if (relativePath.endsWith(".jpg") || relativePath.endsWith(".png") ||
206 relativePath.endsWith(".jpeg") || relativePath.endsWith(".gif"))
207 {
208 // Likely an image file - add assets/ prefix if needed
209 result = RemoteResources::BASE_URL + RemoteResources::ASSETS_PATH + relativePath;
210 }
211 else
212 {
213 // Default case - just append to base URL
214 result = RemoteResources::BASE_URL + relativePath;
215 }
216
217 return result;
218 }

469 void refreshSubItems()
470 {
471 clearSubItems();
472
473 // No owner means we can't properly populate
474 if (owner == nullptr)
475 return;
476
477 if (itemType == ItemType::Root)
478 {
479 // Add Recently Used group
480 addSubItem(new GearTreeItem(ItemType::RecentlyUsed, "Recently Used", owner));
481
482 // Add Favorites group
483 addSubItem(new GearTreeItem(ItemType::Favorites, "My Gear", owner));
484
485 // Add Categories group
486 addSubItem(new GearTreeItem(ItemType::Category, "Categories", owner));
487 }
488 else if (itemType == ItemType::RecentlyUsed)
489 {
490 // Get recently used items from cache
491 CacheManager &cache = CacheManager::getInstance();
492 juce::StringArray recentlyUsed = cache.getRecentlyUsed(CacheManager::MAX_RECENTLY_USED); // Show up to MAX_RECENTLY_USED items
493
494 if (!recentlyUsed.isEmpty())
495 {
496 // Get all items from the library
497 const auto &items = owner->getItems();
498
499 // Add each recently used item
500 for (const auto &unitId : recentlyUsed)
501 {
502 // Find the gear item in the library
503 for (int i = 0; i < items.size(); ++i)
504 {
505 const auto &item = items.getReference(i);
506 if (item.unitId == unitId)
507 {
508 addSubItem(new GearTreeItem(ItemType::Gear, item.name, owner,
509 const_cast<GearItem *>(&item), i));
510 break;
511 }
512 }
513 }
514 }
515 }
516 else if (itemType == ItemType::Category && name == "Categories")
517 {
518 // Get all items from the library
519 const auto &items = owner->getItems();
520
521 // Collect unique categories from the items
522 juce::StringArray categories;
523
524 // First pass - gather all unique categories
525 for (int i = 0; i < items.size(); ++i)
526 {
527 const auto &item = items.getReference(i);
528
529 // Get the category string or derive it from the enum
530 juce::String category;
531
532 if (!item.categoryString.isEmpty())
533 {
534 category = item.categoryString.toLowerCase();
535 }
536 else
537 {
538 // Fallback to enum if string is empty
539 switch (item.category)
540 {
541 case GearCategory::EQ:
542 category = "equalizer";
543 break;
544 case GearCategory::Compressor:
545 category = "compressor";
546 break;
547 case GearCategory::Preamp:
548 category = "preamp";
549 break;
550 case GearCategory::Other:
551 category = "other";
552 break;
553 }
554 }
555
556 // Add to our unique list if not already present
557 if (!category.isEmpty() && !categories.contains(category))
558 categories.add(category);
559 }
560
561 // Sort categories alphabetically
562 categories.sort(true);
563
564 // Add a tree item for each category
565 for (const auto &category : categories)
566 {
567 // Format category name for display (capitalize first letter)
568 juce::String displayName = category.substring(0, 1).toUpperCase() + category.substring(1);
569 addSubItem(new GearTreeItem(ItemType::Category, displayName, owner));
570 }
571 }
572 else if (itemType == ItemType::Category)
573 {
574 // Find items matching this category
575 const auto &items = owner->getItems();
576 bool hasItems = false;
577
578 for (int i = 0; i < items.size(); ++i)
579 {
580 const auto &item = items.getReference(i);
581
582 // Check if this item matches the category
583 bool matches = false;
584
585 if (name.equalsIgnoreCase("EQ") || name.equalsIgnoreCase("Equalizer"))
586 {
587 matches = (item.category == GearCategory::EQ ||
588 item.categoryString.equalsIgnoreCase("equalizer") ||
589 item.categoryString.equalsIgnoreCase("eq"));
590 }
591 else if (name.equalsIgnoreCase("Compressor"))
592 {
593 matches = (item.category == GearCategory::Compressor ||
594 item.categoryString.equalsIgnoreCase("compressor"));
595 }
596 else if (name.equalsIgnoreCase("Preamp"))
597 {
598 matches = (item.category == GearCategory::Preamp ||
599 item.categoryString.equalsIgnoreCase("preamp"));
600 }
601 else if (name.equalsIgnoreCase("Other"))
602 {
603 matches = (item.category == GearCategory::Other ||
604 item.categoryString.equalsIgnoreCase("other"));
605 }
606 else
607 {
608 // For any other category, match directly with the item's categoryString
609 matches = item.categoryString.equalsIgnoreCase(name);
610 }
611
612 if (matches)
613 {
614 addSubItem(new GearTreeItem(ItemType::Gear, item.name, owner,
615 const_cast<GearItem *>(&item), i));
616 hasItems = true;
617 }
618 }
619
620 // If no items were found for this category, don't expand it
621 if (!hasItems)
622 setOpen(false);
623 }
624 else if (itemType == ItemType::Favorites)
625 {
626 // Get favorite items from cache
627 CacheManager &cache = CacheManager::getInstance();
628 juce::StringArray favorites = cache.getFavorites();
629
630 if (!favorites.isEmpty())
631 {
632 // Get all items from the library
633 const auto &items = owner->getItems();
634
635 // Group favorites by category (same approach as Categories tree)
636 juce::HashMap<juce::String, juce::Array<GearItem *>> categoryGroups;
637
638 // Collect favorite items and group them by category
639 for (const auto &unitId : favorites)
640 {
641 // Find the gear item in the library
642 for (int i = 0; i < items.size(); ++i)
643 {
644 const auto &item = items.getReference(i);
645 if (item.unitId == unitId)
646 {
647 // Get the category string or derive it from the enum
648 juce::String category;
649 if (!item.categoryString.isEmpty())
650 {
651 category = item.categoryString.toLowerCase();
652 }
653 else
654 {
655 // Fallback to enum if string is empty
656 switch (item.category)
657 {
658 case GearCategory::EQ:
659 category = "equalizer";
660 break;
661 case GearCategory::Compressor:
662 category = "compressor";
663 break;
664 case GearCategory::Preamp:
665 category = "preamp";
666 break;
667 case GearCategory::Other:
668 category = "other";
669 break;
670 }
671 }
672
673 if (!categoryGroups.contains(category))
674 categoryGroups.set(category, juce::Array<GearItem *>());
675
676 categoryGroups.getReference(category).add(const_cast<GearItem *>(&item));
677 break;
678 }
679 }
680 }
681
682 // Add category groups as sub-items
683 for (auto it = categoryGroups.begin(); it != categoryGroups.end(); ++it)
684 {
685 juce::String categoryName = it.getKey();
686 juce::String displayName = categoryName.substring(0, 1).toUpperCase() + categoryName.substring(1);
687
688 auto categoryNode = new GearTreeItem(ItemType::Category, displayName, owner);
689 addSubItem(categoryNode);
690
691 // Add gear items to this category group
692 for (auto *item : it.getValue())
693 {
694 // Find the index of this item in the original array
695 int itemIndex = -1;
696 for (int i = 0; i < items.size(); ++i)
697 {
698 if (&items.getReference(i) == item)
699 {
700 itemIndex = i;
701 break;
702 }
703 }
704
705 if (itemIndex >= 0)
706 {
707 categoryNode->addSubItem(new GearTreeItem(ItemType::Gear, item->name, owner,
708 item, itemIndex));
709 }
710 }
711 }
712 }
713 }

766 juce::PopupMenu menu;
767 menu.addItem(1, "Clear My Gear");
768
769 menu.showMenuAsync(juce::PopupMenu::Options(),
770 [this](int result)
771 {
772 if (result == 1 && owner != nullptr)
773 {
774 owner->clearFavorites();
775 }
776 });
777 return;
778 }
779
780 // Handle star icon clicks for gear items
781 if (itemType == ItemType::Gear && gearItem != nullptr && e.mods.isLeftButtonDown())
782 {
783 // Check if click is in the left portion of the item where the star would be
784 // The star is drawn at the left edge, so check if click is in the left 20 pixels
785 const int starAreaWidth = 20;
786
787 if (e.x <= starAreaWidth)
788 {
789 // Toggle favorite status
790 CacheManager &cache = CacheManager::getInstance();
791 bool isFavorite = cache.isFavorite(gearItem->unitId);
792
793 if (isFavorite)
794 {
795 cache.removeFromFavorites(gearItem->unitId);
796 }
797 else
798 {
799 cache.addToFavorites(gearItem->unitId);
800 }
801
802 // Refresh only the favorites section to preserve tree expansion state
803 if (owner != nullptr)
804 {
805 owner->refreshFavoritesSection();
806 }
807
808 return; // Don't proceed with other operations
809 }
810 }
811
812 // Handle default behavior (expand/collapse) for all other clicks
813 TreeViewItem::itemClicked(e);
814
815 // If this is a gear item, make it draggable
816 if (itemType == ItemType::Gear && gearItem != nullptr && e.mods.isLeftButtonDown())
817 {
818 // Find the parent drag container
819 juce::Component *comp = getOwnerView();
820 if (comp == nullptr)
821 return;
822
823 juce::DragAndDropContainer *container = juce::DragAndDropContainer::findParentDragContainerFor(comp);
