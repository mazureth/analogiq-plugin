206 static juce::String getFullUrl(const juce::String &relativePath)
207 {
208 // If already a full URL, return as is
209 if (relativePath.startsWith("http"))
210 return relativePath;
211
212 // If this is an absolute path on the filesystem, return as is
213 if (relativePath.startsWith("/"))
214 return relativePath;
215
216 // Handle the case where we might need to add assets/ or units/ prefix
217 juce::String result;
218
219 if (relativePath.startsWith("assets/") || relativePath.startsWith("units/"))
220 {
221 // Path already has the correct folder prefix
222 result = RemoteResources::BASE_URL + relativePath;
223 }
224 else if (relativePath.endsWith(".json"))
225 {
226 // Likely a schema file - add units/ prefix if needed
227 result = RemoteResources::BASE_URL + RemoteResources::SCHEMAS_PATH + relativePath;
228 }
229 else if (relativePath.endsWith(".jpg") || relativePath.endsWith(".png") ||
230 relativePath.endsWith(".jpeg") || relativePath.endsWith(".gif"))
231 {
232 // Likely an image file - add assets/ prefix if needed
233 result = RemoteResources::BASE_URL + RemoteResources::ASSETS_PATH + relativePath;
234 }
235 else
236 {
237 // Default case - just append to base URL
238 result = RemoteResources::BASE_URL + relativePath;
239 }
240
241 return result;
242 }

423 void refreshSubItems()
424 {
425 clearSubItems();
426
427 // No owner means we can't properly populate
428 if (owner == nullptr)
429 return;
430
431 if (itemType == ItemType::Root)
432 {
433 // Add Categories group
434 addSubItem(new GearTreeItem(ItemType::Category, "Categories", owner));
435 }
436 else if (itemType == ItemType::Category && name == "Categories")
437 {
438 // Get all items from the library
439 const auto &items = owner->getItems();
440
441 // Collect unique categories from the items
442 juce::StringArray categories;
443
444 // First pass - gather all unique categories
445 for (int i = 0; i < items.size(); ++i)
446 {
447 const auto &item = items.getReference(i);
448
449 // Get the category string or derive it from the enum
450 juce::String category;
451
452 if (!item.categoryString.isEmpty())
453 {
454 category = item.categoryString.toLowerCase();
455 }
456 else
457 {
458 // Fallback to enum if string is empty
459 switch (item.category)
460 {
461 case GearCategory::EQ:
462 category = "equalizer";
463 break;
464 case GearCategory::Compressor:
465 category = "compressor";
466 break;
467 case GearCategory::Preamp:
468 category = "preamp";
469 break;
470 case GearCategory::Other:
471 category = "other";
472 break;
473 }
474 }
475
476 // Add to our unique list if not already present
477 if (!category.isEmpty() && !categories.contains(category))
478 categories.add(category);
479 }
480
481 // Sort categories alphabetically
482 categories.sort(true);
483
484 // Add a tree item for each category
485 for (const auto &category : categories)
486 {
487 // Format category name for display (capitalize first letter)
488 juce::String displayName = category.substring(0, 1).toUpperCase() + category.substring(1);
489 addSubItem(new GearTreeItem(ItemType::Category, displayName, owner));
490 }
491 }
492 else if (itemType == ItemType::Category)
493 {
494 // Find items matching this category
495 const auto &items = owner->getItems();
496 bool hasItems = false;
497
498 for (int i = 0; i < items.size(); ++i)
499 {
500 const auto &item = items.getReference(i);
501
502 // Check if this item matches the category
503 bool matches = false;
504
505 if (name.equalsIgnoreCase("EQ") || name.equalsIgnoreCase("Equalizer"))
506 {
507 matches = (item.category == GearCategory::EQ ||
508 item.categoryString.equalsIgnoreCase("equalizer") ||
509 item.categoryString.equalsIgnoreCase("eq"));
510 }
511 else if (name.equalsIgnoreCase("Compressor"))
512 {
513 matches = (item.category == GearCategory::Compressor ||
514 item.categoryString.equalsIgnoreCase("compressor"));
515 }
516 else if (name.equalsIgnoreCase("Preamp"))
517 {
518 matches = (item.category == GearCategory::Preamp ||
519 item.categoryString.equalsIgnoreCase("preamp"));
520 }
521 else if (name.equalsIgnoreCase("Other"))
522 {
523 matches = (item.category == GearCategory::Other ||
524 item.categoryString.equalsIgnoreCase("other"));
525 }
526 else
527 {
528 // For any other category, match directly with the item's categoryString
529 matches = item.categoryString.equalsIgnoreCase(name);
530 }
531
532 if (matches)
533 {
534 addSubItem(new GearTreeItem(ItemType::Gear, item.name, owner,
535 const_cast<GearItem *>(&item), i));
536 hasItems = true;
537 }
538 }
539
540 // If no items were found for this category, don't expand it
541 if (!hasItems)
542 setOpen(false);
543 }

574 void itemClicked(const juce::MouseEvent &e) override
575 {
576 // First handle the default behavior
577 TreeViewItem::itemClicked(e);

579 // If this is a gear item, make it draggable

581 {
582 // Find the parent drag container
583 juce::Component *comp = getOwnerView();
584 if (comp == nullptr)
585 return;
586
587 juce::DragAndDropContainer *container = juce::DragAndDropContainer::findParentDragContainerFor(comp);
588 if (container == nullptr)
589 return;
590
591 // Create a custom drag image
592 int itemWidth = 150;
593 int itemHeight = 40;
594
595 juce::Image dragImage(juce::Image::ARGB, itemWidth, itemHeight, true);
596 juce::Graphics g(dragImage);
597
598 g.setColour(juce::Colours::darkgrey);
599 g.fillRoundedRectangle(0.0f, 0.0f, (float)itemWidth, (float)itemHeight, 8.0f);
600
601 // Add a visual indicator
602 g.setColour(juce::Colours::greenyellow);
603 g.fillEllipse(10, itemHeight / 2 - 6, 12, 12);
604
605 g.setColour(juce::Colours::white);
606 g.setFont(14.0f);
607 g.drawText(gearItem->name, 30, 5, itemWidth - 40, 30, juce::Justification::centredLeft);
608
609 // Create a structured drag description that the rack can recognize
610 juce::String dragDesc = "GEAR:" + juce::String(gearIndex) + ":" + gearItem->name;
611

613 juce::Point<int> imageOffset(e.x - 10, e.y - itemHeight / 2);
