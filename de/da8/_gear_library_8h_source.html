202 static juce::String getFullUrl(const juce::String &relativePath)
203 {
204 // If already a full URL, return as is
205 if (relativePath.startsWith("http"))
206 return relativePath;
207
208 // If this is an absolute path on the filesystem, return as is
209 if (relativePath.startsWith("/"))
210 return relativePath;
211
212 // Handle the case where we might need to add assets/ or units/ prefix
213 juce::String result;
214
215 if (relativePath.startsWith("assets/") || relativePath.startsWith("units/"))
216 {
217 // Path already has the correct folder prefix
218 result = RemoteResources::BASE_URL + relativePath;
219 }
220 else if (relativePath.endsWith(".json"))
221 {
222 // Likely a schema file - add units/ prefix if needed
223 result = RemoteResources::BASE_URL + RemoteResources::SCHEMAS_PATH + relativePath;
224 }
225 else if (relativePath.endsWith(".jpg") || relativePath.endsWith(".png") ||
226 relativePath.endsWith(".jpeg") || relativePath.endsWith(".gif"))
227 {
228 // Likely an image file - add assets/ prefix if needed
229 result = RemoteResources::BASE_URL + RemoteResources::ASSETS_PATH + relativePath;
230 }
231 else
232 {
233 // Default case - just append to base URL
234 result = RemoteResources::BASE_URL + relativePath;
235 }
236
237 return result;
238 }

513 void refreshSubItems()
514 {
515 clearSubItems();
516
517 // No owner means we can't properly populate
518 if (owner == nullptr)
519 return;
520
521 if (type == ItemType::Root)
522 {
523 // Add Recently Used group
524 addSubItem(new GearTreeItem(ItemType::RecentlyUsed, "Recently Used", owner, &owner->getCacheManager()));
525
526 // Add Favorites group
527 addSubItem(new GearTreeItem(ItemType::Favorites, "My Gear", owner, &owner->getCacheManager()));
528
529 // Add Categories group
530 addSubItem(new GearTreeItem(ItemType::Category, "Categories", owner, &owner->getCacheManager()));
531 }
532 else if (type == ItemType::RecentlyUsed)
533 {
534 // Get recently used items from cache
535 if (cacheManager != nullptr)
536 {
537 juce::StringArray recentlyUsed = cacheManager->getRecentlyUsed(CacheManager::MAX_RECENTLY_USED); // Show up to MAX_RECENTLY_USED items
538
539 if (!recentlyUsed.isEmpty())
540 {
541 // Get all items from the library
542 const auto &items = owner->getItems();
543
544 // Add each recently used item
545 for (const auto &unitId : recentlyUsed)
546 {
547 // Find the gear item in the library
548 for (int i = 0; i < items.size(); ++i)
549 {
550 const auto &item = items.getReference(i);
551 if (item.unitId == unitId)
552 {
553 addSubItem(new GearTreeItem(ItemType::Gear, item.name, owner, cacheManager,
554 const_cast<GearItem *>(&item), i));
555 break;
556 }
557 }
558 }
559 }
560 }
561 }
562 else if (type == ItemType::Category && name == "Categories")
563 {
564 // Get all items from the library
565 const auto &items = owner->getItems();
566
567 // Collect unique categories from the items
568 juce::StringArray categories;
569
570 // First pass - gather all unique categories
571 for (int i = 0; i < items.size(); ++i)
572 {
573 const auto &item = items.getReference(i);
574 if (!categories.contains(item.categoryString))
575 categories.add(item.categoryString);
576 }
577
578 // Add a GearTreeItem for each category
579 for (const auto &category : categories)
580 {
581 // Format category name for display (capitalize first letter)
582 juce::String displayName = category.substring(0, 1).toUpperCase() + category.substring(1);
583 addSubItem(new GearTreeItem(ItemType::Category, displayName, owner, &owner->getCacheManager()));
584 }
585 }
586 else if (type == ItemType::Category)
587 {
588 // Get all items from the library
589 const auto &items = owner->getItems();
590 bool hasItems = false;
591 for (int i = 0; i < items.size(); ++i)
592 {
593 const auto &item = items.getReference(i);
594 if (item.categoryString.equalsIgnoreCase(name))
595 {
596 addSubItem(new GearTreeItem(ItemType::Gear, item.name, owner, &owner->getCacheManager(),
597 const_cast<GearItem *>(&item), i));
598 hasItems = true;
599 }
600 }
601 if (!hasItems)
602 {
603 addSubItem(new GearTreeItem(ItemType::Message, "No items in this category", owner, &owner->getCacheManager()));
604 }
605 }
606 else if (type == ItemType::Favorites)
607 {
608 // Favorites are handled by refreshFavoritesSection() in GearLibrary.cpp
609 // This method is not used for favorites
610 }

641 void itemClicked(const juce::MouseEvent &e) override
642 {
643 // Handle right-click on Recently Used item
644 if (type == ItemType::RecentlyUsed && e.mods.isRightButtonDown())
645 {
646 juce::PopupMenu menu;
647 menu.addItem(1, "Clear Recently Used");
648
649 menu.showMenuAsync(juce::PopupMenu::Options(),
650 [this](int result)
651 {
652 if (result == 1 && owner != nullptr)
653 {
654 owner->clearRecentlyUsed();
655 }
656 });
657 return;
658 }
659
660 // Handle right-click on Favorites item
661 if (type == ItemType::Favorites && e.mods.isRightButtonDown())
662 {
663 juce::PopupMenu menu;
664 menu.addItem(1, "Clear My Gear");
665
666 menu.showMenuAsync(juce::PopupMenu::Options(),
667 [this](int result)
668 {
669 if (result == 1 && owner != nullptr)
670 {
671 owner->clearFavorites();
672 }
673 });
674 return;
675 }
676
677 // Handle star icon clicks for gear items
678 if (type == ItemType::Gear && gearItem != nullptr && e.mods.isLeftButtonDown())
679 {
680 // Check if click is in the left portion of the item where the star would be
681 // The star is drawn at the left edge, so check if click is in the left 20 pixels
682 const int starAreaWidth = 20;
683
684 if (e.x <= starAreaWidth)
685 {
686 // Toggle favorite status
687 if (cacheManager != nullptr)
688 {
689 bool isFavorite = cacheManager->isFavorite(gearItem->unitId);
690
691 if (isFavorite)
692 {
693 cacheManager->removeFromFavorites(gearItem->unitId);
694 }
695 else
696 {
697 cacheManager->addToFavorites(gearItem->unitId);
698 }
699 }
700
701 // Refresh only the favorites section to preserve tree expansion state
702 if (owner != nullptr)
703 {
704 owner->refreshFavoritesSection();
705 }
706
707 return; // Don't proceed with other operations
708 }
709 }
710
711 // Handle default behavior (expand/collapse) for all other clicks
712 TreeViewItem::itemClicked(e);
713
714 // If this is a gear item, make it draggable
715 if (type == ItemType::Gear && gearItem != nullptr && e.mods.isLeftButtonDown())
716 {
717 // Find the parent drag container
718 juce::Component *comp = getOwnerView();
719 if (comp == nullptr)
720 return;
721
722 juce::DragAndDropContainer *container = juce::DragAndDropContainer::findParentDragContainerFor(comp);
723 if (container == nullptr)
724 return;
725
726 // Create a custom drag image
727 int itemWidth = 150;
728 int itemHeight = 40;
729
730 juce::Image dragImage(juce::Image::ARGB, itemWidth, itemHeight, true);
731 juce::Graphics g(dragImage);
732
733 g.setColour(juce::Colours::darkgrey);
734 g.fillRoundedRectangle(0.0f, 0.0f, (float)itemWidth, (float)itemHeight, 8.0f);
735
736 // Add a visual indicator
737 g.setColour(juce::Colours::greenyellow);
738 g.fillEllipse(10, itemHeight / 2 - 6, 12, 12);
739
740 g.setColour(juce::Colours::white);
741 g.setFont(14.0f);
742 g.drawText(gearItem->name, 30, 5, itemWidth - 40, 30, juce::Justification::centredLeft);
743
744 // Create a structured drag description that the rack can recognize
745 juce::String dragDesc = "GEAR:" + juce::String(itemIndex) + ":" + gearItem->name;
746
747 // Calculate the drag image offset from the mouse
748 juce::Point<int> imageOffset(e.x - 10, e.y - itemHeight / 2);
749
750 // Use the newer API to start the drag operation
751 container->startDragging(dragDesc, comp, dragImage, true, &imageOffset, nullptr);
752 }
