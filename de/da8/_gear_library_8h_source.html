194 static juce::String getFullUrl(const juce::String &relativePath)
195 {
196 // If already a full URL, return as is
197 if (relativePath.startsWith("http"))
198 return relativePath;
199
200 // If this is an absolute path on the filesystem, return as is
201 if (relativePath.startsWith("/"))
202 return relativePath;
203
204 // Handle the case where we might need to add assets/ or units/ prefix
205 juce::String result;
206
207 if (relativePath.startsWith("assets/") || relativePath.startsWith("units/"))
208 {
209 // Path already has the correct folder prefix
210 result = RemoteResources::BASE_URL + relativePath;
211 }
212 else if (relativePath.endsWith(".json"))
213 {
214 // Likely a schema file - add units/ prefix if needed
215 result = RemoteResources::BASE_URL + RemoteResources::SCHEMAS_PATH + relativePath;
216 }
217 else if (relativePath.endsWith(".jpg") || relativePath.endsWith(".png") ||
218 relativePath.endsWith(".jpeg") || relativePath.endsWith(".gif"))
219 {
220 // Likely an image file - add assets/ prefix if needed
221 result = RemoteResources::BASE_URL + RemoteResources::ASSETS_PATH + relativePath;
222 }
223 else
224 {
225 // Default case - just append to base URL
226 result = RemoteResources::BASE_URL + relativePath;
227 }
228
229 return result;
230 }

505 void refreshSubItems()
506 {
507 clearSubItems();
508
509 // No owner means we can't properly populate
510 if (owner == nullptr)
511 return;
512
513 if (type == ItemType::Root)
514 {
515 // Add Recently Used group
516 addSubItem(new GearTreeItem(ItemType::RecentlyUsed, "Recently Used", owner, &owner->getCacheManager()));
517
518 // Add Favorites group
519 addSubItem(new GearTreeItem(ItemType::Favorites, "My Gear", owner, &owner->getCacheManager()));
520
521 // Add Categories group
522 addSubItem(new GearTreeItem(ItemType::Category, "Categories", owner, &owner->getCacheManager()));
523 }
524 else if (type == ItemType::RecentlyUsed)
525 {
526 // Get recently used items from cache
527 if (cacheManager != nullptr)
528 {
529 juce::StringArray recentlyUsed = cacheManager->getRecentlyUsed(CacheManager::MAX_RECENTLY_USED); // Show up to MAX_RECENTLY_USED items
530
531 if (!recentlyUsed.isEmpty())
532 {
533 // Get all items from the library
534 const auto &items = owner->getItems();
535
536 // Add each recently used item
537 for (const auto &unitId : recentlyUsed)
538 {
539 // Find the gear item in the library
540 for (int i = 0; i < items.size(); ++i)
541 {
542 const auto &item = items.getReference(i);
543 if (item.unitId == unitId)
544 {
545 addSubItem(new GearTreeItem(ItemType::Gear, item.name, owner, cacheManager,
546 const_cast<GearItem *>(&item), i));
547 break;
548 }
549 }
550 }
551 }
552 }
553 }
554 else if (type == ItemType::Category && name == "Categories")
555 {
556 // Get all items from the library
557 const auto &items = owner->getItems();
558
559 // Collect unique categories from the items
560 juce::StringArray categories;
561
562 // First pass - gather all unique categories
563 for (int i = 0; i < items.size(); ++i)
564 {
565 const auto &item = items.getReference(i);
566 if (!categories.contains(item.categoryString))
567 categories.add(item.categoryString);
568 }
569
570 // Add a GearTreeItem for each category
571 for (const auto &category : categories)
572 {
573 // Format category name for display (capitalize first letter)
574 juce::String displayName = category.substring(0, 1).toUpperCase() + category.substring(1);
575 addSubItem(new GearTreeItem(ItemType::Category, displayName, owner, &owner->getCacheManager()));
576 }
577 }
578 else if (type == ItemType::Category)
579 {
580 // Get all items from the library
581 const auto &items = owner->getItems();
582 bool hasItems = false;
583 for (int i = 0; i < items.size(); ++i)
584 {
585 const auto &item = items.getReference(i);
586 if (item.categoryString.equalsIgnoreCase(name))
587 {
588 addSubItem(new GearTreeItem(ItemType::Gear, item.name, owner, &owner->getCacheManager(),
589 const_cast<GearItem *>(&item), i));
590 hasItems = true;
591 }
592 }
593 if (!hasItems)
594 {
595 addSubItem(new GearTreeItem(ItemType::Message, "No items in this category", owner, &owner->getCacheManager()));
596 }
597 }
598 else if (type == ItemType::Favorites)
599 {
600 // Favorites are handled by refreshFavoritesSection() in GearLibrary.cpp
601 // This method is not used for favorites
602 }

633 void itemClicked(const juce::MouseEvent &e) override
634 {
635 // Handle right-click on Recently Used item
636 if (type == ItemType::RecentlyUsed && e.mods.isRightButtonDown())
637 {
638 juce::PopupMenu menu;
639 menu.addItem(1, "Clear Recently Used");
640
641 menu.showMenuAsync(juce::PopupMenu::Options(),
642 [this](int result)
643 {
644 if (result == 1 && owner != nullptr)
645 {
646 owner->clearRecentlyUsed();
647 }
648 });
649 return;
650 }
651
652 // Handle right-click on Favorites item
653 if (type == ItemType::Favorites && e.mods.isRightButtonDown())
654 {
655 juce::PopupMenu menu;
656 menu.addItem(1, "Clear My Gear");
657
658 menu.showMenuAsync(juce::PopupMenu::Options(),
659 [this](int result)
660 {
661 if (result == 1 && owner != nullptr)
662 {
663 owner->clearFavorites();
664 }
665 });
666 return;
667 }
668
669 // Handle star icon clicks for gear items
670 if (type == ItemType::Gear && gearItem != nullptr && e.mods.isLeftButtonDown())
671 {
672 // Check if click is in the left portion of the item where the star would be
673 // The star is drawn at the left edge, so check if click is in the left 20 pixels
674 const int starAreaWidth = 20;
675
676 if (e.x <= starAreaWidth)
677 {
678 // Toggle favorite status
679 if (cacheManager != nullptr)
680 {
681 bool isFavorite = cacheManager->isFavorite(gearItem->unitId);
682
683 if (isFavorite)
684 {
685 cacheManager->removeFromFavorites(gearItem->unitId);
686 }
687 else
688 {
689 cacheManager->addToFavorites(gearItem->unitId);
690 }
691 }
692
693 // Refresh only the favorites section to preserve tree expansion state
694 if (owner != nullptr)
695 {
696 owner->refreshFavoritesSection();
697 }
698
699 return; // Don't proceed with other operations
700 }
701 }
702
703 // Handle default behavior (expand/collapse) for all other clicks
704 TreeViewItem::itemClicked(e);
705
706 // If this is a gear item, make it draggable
707 if (type == ItemType::Gear && gearItem != nullptr && e.mods.isLeftButtonDown())
708 {
709 // Find the parent drag container
710 juce::Component *comp = getOwnerView();
711 if (comp == nullptr)
712 return;
713
714 juce::DragAndDropContainer *container = juce::DragAndDropContainer::findParentDragContainerFor(comp);
715 if (container == nullptr)
716 return;
717
718 // Create a custom drag image
719 int itemWidth = 150;
720 int itemHeight = 40;
721
722 juce::Image dragImage(juce::Image::ARGB, itemWidth, itemHeight, true);
723 juce::Graphics g(dragImage);
724
725 g.setColour(juce::Colours::darkgrey);
726 g.fillRoundedRectangle(0.0f, 0.0f, (float)itemWidth, (float)itemHeight, 8.0f);
727
728 // Add a visual indicator
729 g.setColour(juce::Colours::greenyellow);
730 g.fillEllipse(10, itemHeight / 2 - 6, 12, 12);
731
732 g.setColour(juce::Colours::white);
733 g.setFont(14.0f);
734 g.drawText(gearItem->name, 30, 5, itemWidth - 40, 30, juce::Justification::centredLeft);
735
736 // Create a structured drag description that the rack can recognize
737 juce::String dragDesc = "GEAR:" + juce::String(itemIndex) + ":" + gearItem->name;
738
739 // Calculate the drag image offset from the mouse
740 juce::Point<int> imageOffset(e.x - 10, e.y - itemHeight / 2);
741
742 // Use the newer API to start the drag operation
743 container->startDragging(dragDesc, comp, dragImage, true, &imageOffset, nullptr);
744 }
