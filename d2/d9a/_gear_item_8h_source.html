287
288 // Clear images in controls with validation
289 for (auto &control : controls)
290 {
291 if (control.loadedImage.isValid())
292 control.loadedImage = juce::Image();
293 if (control.switchSpriteSheet.isValid())
294 control.switchSpriteSheet = juce::Image();
295 if (control.faderImage.isValid())
296 control.faderImage = juce::Image();
297 if (control.buttonSpriteSheet.isValid())
298 control.buttonSpriteSheet = juce::Image();
299 }
300
301 // Clear the controls array completely to release any internal references
302 controls.clear();
303
304 // Clear string arrays that might hold references
305 tags.clear();

307 // Force a small delay to allow JUCE internal cleanup
308 juce::Thread::sleep(1);
