21bool GearItem::loadImage()
22{
23 // If no thumbnail specified, use placeholder
24 if (thumbnailImage.isEmpty())
25 return createPlaceholderImage();
26
27 // Extract filename from thumbnail path
28 juce::String filename = juce::File(thumbnailImage).getFileName();
29
30 // Check cache first
31 CacheManager &cache = CacheManager::getInstance();
32 if (cache.isThumbnailCached(unitId, filename))
33 {
34 juce::Image cachedImage = cache.loadThumbnailFromCache(unitId, filename);
35 if (cachedImage.isValid())
36 {
37 image = cachedImage;
38 return true;
39 }
40 }
41
42 // Check if the thumbnail is a remote path
43 if (thumbnailImage.startsWith("assets/") || thumbnailImage.startsWith("http"))
44 {
45 // Determine the full URL using the helper method
46 juce::String imageUrl = GearLibrary::getFullUrl(thumbnailImage);
47
48 // Create URL object for the image
49 juce::URL url(imageUrl);
50
51 // Try to download the image using the network fetcher
52 bool success = false;
53 juce::MemoryBlock imageData = networkFetcher.fetchBinaryBlocking(url, success);
54
55 if (success && imageData.getSize() > 0)
56 {
57 // Create image from the memory block
58 juce::MemoryInputStream inputStream(imageData, false);
59 juce::JPEGImageFormat jpegFormat;
60 juce::PNGImageFormat pngFormat;
61
62 // Try to load as JPEG first, then PNG
63 if (jpegFormat.canUnderstand(inputStream))
64 {
65 inputStream.setPosition(0);
66 image = jpegFormat.decodeImage(inputStream);
67 }
68 else
69 {
70 inputStream.setPosition(0);
71 if (pngFormat.canUnderstand(inputStream))
72 {
73 inputStream.setPosition(0);
74 image = pngFormat.decodeImage(inputStream);
75 }
76 }
77
78 // If successfully loaded image, cache it and return true
79 if (image.isValid())
80 {
81 cache.saveThumbnailToCache(unitId, filename, image);
82 return true;
83 }
84 }
85 }
86
87 // If we get here, loading the actual image failed, so create a placeholder
88 return createPlaceholderImage();
89}

99bool GearItem::createPlaceholderImage()
100{
101 // Create a placeholder colored image based on category
102 image = juce::Image(juce::Image::ARGB, 24, 24, true);
103 juce::Graphics g(image);
104
105 // Use different colors for different categories
106 switch (category)
107 {
108 case GearCategory::EQ:
109 g.setColour(juce::Colours::orange);
110 break;
111 case GearCategory::Preamp:
112 g.setColour(juce::Colours::red);
113 break;
114 case GearCategory::Compressor:
115 g.setColour(juce::Colours::blue);
116 break;
117 default:
118 g.setColour(juce::Colours::green);
119 break;
120 }
121
122 // Draw a rounded rectangle for the thumbnail
123 g.fillRoundedRectangle(0.0f, 0.0f, 24.0f, 24.0f, 4.0f);
124
125 // Draw the first letter of the gear name
126 g.setColour(juce::Colours::white);
127 g.setFont(16.0f);
128 g.drawText(name.substring(0, 1).toUpperCase(),
129 0, 0, 24, 24, juce::Justification::centred);
130
131 return true;
132}

142void GearItem::createInstance(const juce::String &sourceUnitId)
143{
144 // Store current state

154

156 if (wasInstance)
157 {
158 // Restore our current control values
159 controls = currentControls;
160 // Don't update initial values when recreating an instance
161 }
162 else

164 // For new instances, set initial values to current values
165 for (auto &control : controls)
166 {
167 control.initialValue = control.value;

205void GearItem::saveToJSON(juce::File destinationFile)
206{
207 // Create a JSON object with all of our properties
208 juce::DynamicObject::Ptr jsonObj = new juce::DynamicObject();
209
210 // Add all properties to the JSON object
211 jsonObj->setProperty("unitId", unitId);
212 jsonObj->setProperty("name", name);
213 jsonObj->setProperty("manufacturer", manufacturer);
214 jsonObj->setProperty("category", categoryString);
215 jsonObj->setProperty("version", version);
216 jsonObj->setProperty("schemaPath", schemaPath);
217 jsonObj->setProperty("thumbnailImage", thumbnailImage);
218
219 // Add instance management properties
220 jsonObj->setProperty("isInstance", isInstance);
221 jsonObj->setProperty("instanceId", instanceId);
222 jsonObj->setProperty("sourceUnitId", sourceUnitId);
223
224 // Convert GearType enum to string
225 juce::String typeStr;
226 switch (type)
227 {
228 case GearType::Series500:
229 typeStr = "500Series";
230 break;
231 case GearType::Rack19Inch:
232 typeStr = "Rack19Inch";
233 break;
234 case GearType::UserCreated:
235 typeStr = "UserCreated";
236 break;
237 case GearType::Other:
238 typeStr = "Other";
239 break;
240 }
241 jsonObj->setProperty("type", typeStr);
242
243 // Add slot size
244 jsonObj->setProperty("slotSize", slotSize);
245
246 // Add tags as an array
247 juce::Array<juce::var> tagsArray;
248 for (const auto &tag : tags)
249 tagsArray.add(tag);
250 jsonObj->setProperty("tags", tagsArray);
251
252 // Add controls as an array
253 juce::Array<juce::var> controlsArray;
254 for (const auto &control : controls)
255 {
256 juce::DynamicObject::Ptr controlObj = new juce::DynamicObject();
257
258 // Convert control type to string
259 juce::String controlTypeStr;
260 switch (control.type)
261 {
262 case GearControl::Type::Button:
263 controlTypeStr = "Button";
264 break;
265 case GearControl::Type::Fader:
266 controlTypeStr = "Fader";
267 break;
268 case GearControl::Type::Switch:
269 controlTypeStr = "Switch";
270 break;
271 }
272
273 controlObj->setProperty("type", controlTypeStr);
274 controlObj->setProperty("name", control.name);
275
276 // Convert position rectangle to JSON
277 juce::DynamicObject::Ptr posObj = new juce::DynamicObject();
278 posObj->setProperty("x", control.position.getX());
279 posObj->setProperty("y", control.position.getY());
280 posObj->setProperty("width", control.position.getWidth());
281 posObj->setProperty("height", control.position.getHeight());
282
283 controlObj->setProperty("position", posObj.get());
284 controlObj->setProperty("value", control.value);
285
286 controlsArray.add(controlObj.get());
287 }
288
289 jsonObj->setProperty("controls", controlsArray);
290
291 // Convert to JSON string
292 juce::var jsonVar(jsonObj.get());
293 juce::String jsonString = juce::JSON::toString(jsonVar);
294
295 // Write to file
296 destinationFile.replaceWithText(jsonString);
297}

309GearItem GearItem::loadFromJSON(juce::File sourceFile, INetworkFetcher &networkFetcher)
310{
311 // Read the JSON from file
312 juce::String jsonString = sourceFile.loadFileAsString();
313
314 // Parse the JSON
315 juce::var jsonVar = juce::JSON::parse(jsonString);
316
317 // Check if we have a valid JSON object
318 if (!jsonVar.isObject())
319 throw std::runtime_error("Invalid JSON format in gear item file");
320
321 // Extract fields from JSON
322 juce::String unitId = jsonVar.getProperty("unitId", "");
323 juce::String name = jsonVar.getProperty("name", "");
324 juce::String manufacturer = jsonVar.getProperty("manufacturer", "");
325 juce::String categoryStr = jsonVar.getProperty("category", "");
326 juce::String version = jsonVar.getProperty("version", "1.0.0");
327 juce::String schemaPath = jsonVar.getProperty("schemaPath", "");
328 juce::String thumbnailImage = jsonVar.getProperty("thumbnailImage", "");
329
330 // Parse tags
331 juce::StringArray tags;
332 if (jsonVar.hasProperty("tags") && jsonVar["tags"].isArray())
333 {
334 juce::Array<juce::var> *tagsArray = jsonVar["tags"].getArray();
335 for (const auto &tag : *tagsArray)
336 tags.add(tag.toString());
337 }
338
339 // Determine type
340 GearType type = GearType::Other;
341 juce::String typeStr = jsonVar.getProperty("type", "");
342 if (typeStr == "500Series")
343 type = GearType::Series500;
344 else if (typeStr == "Rack19Inch")
345 type = GearType::Rack19Inch;
346 else if (typeStr == "UserCreated")
347 type = GearType::UserCreated;
348
349 // Determine category
350 GearCategory category = GearCategory::Other;
351 if (categoryStr == "equalizer" || categoryStr == "eq")
352 category = GearCategory::EQ;
353 else if (categoryStr == "compressor")
354 category = GearCategory::Compressor;
355 else if (categoryStr == "preamp")
356 category = GearCategory::Preamp;
357
358 // Get slot size
359 int slotSize = jsonVar.getProperty("slotSize", 1);
360
361 // Parse controls
362 juce::Array<GearControl> controls;
363 if (jsonVar.hasProperty("controls") && jsonVar["controls"].isArray())
364 {
365 juce::Array<juce::var> *controlsArray = jsonVar["controls"].getArray();
366 for (const auto &controlVar : *controlsArray)
367 {
368 if (!controlVar.isObject())
369 continue;
370
371 // Get control type
372 GearControl::Type controlType = GearControl::Type::Button;
373 juce::String controlTypeStr = controlVar.getProperty("type", "Button");
374 if (controlTypeStr == "Fader")
375 controlType = GearControl::Type::Fader;
376 else if (controlTypeStr == "Switch")
377 controlType = GearControl::Type::Switch;
378
379 // Get control name
380 juce::String controlName = controlVar.getProperty("name", "");
381
382 // Get position
383 juce::Rectangle<float> position;
384 if (controlVar.hasProperty("position") && controlVar["position"].isObject())
385 {
386 position.setX(controlVar["position"].getProperty("x", 0.0f));
387 position.setY(controlVar["position"].getProperty("y", 0.0f));
388 position.setWidth(controlVar["position"].getProperty("width", 0.0f));
389 position.setHeight(controlVar["position"].getProperty("height", 0.0f));
390 }
391
392 // Create control and add to array
393 GearControl control(controlType, controlName, position);
394 control.value = controlVar.getProperty("value", 0.0f);
395 controls.add(control);
396 }
397 }
398
399 // Create and return the gear item with the new constructor
400 GearItem item(unitId, name, manufacturer, categoryStr, version, schemaPath,
401 thumbnailImage, tags, networkFetcher, type, category, slotSize, controls);
402
403 // Try to load the image
404 item.loadImage();
405
406 return item;
407}
