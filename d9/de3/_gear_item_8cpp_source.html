23bool GearItem::loadImage()
24{
25 // If image is already loaded, don't reload
26 if (image.isValid())
27 return true;
28
29 // If no thumbnail specified, use placeholder
30 if (thumbnailImage.isEmpty())
31 return createPlaceholderImage();
32
33 // Extract filename from thumbnail path
34 juce::String filename = fileSystem.getFileName(thumbnailImage);
35
36 // Check cache first using the injected cache manager
37 if (cacheManager.isThumbnailCached(unitId, filename))
38 {
39 juce::Image cachedImage = cacheManager.loadThumbnailFromCache(unitId, filename);
40 if (cachedImage.isValid())
41 {
42 image = cachedImage;
43 return true;
44 }
45 }
46
47 // Check if the thumbnail is a remote path
48 if (thumbnailImage.startsWith("assets/") || thumbnailImage.startsWith("http"))
49 {
50 // Determine the full URL using the helper method
51 juce::String imageUrl = GearLibrary::getFullUrl(thumbnailImage);
52
53 // Create URL object for the image
54 juce::URL url(imageUrl);
55
56 // Try to download the image using the network fetcher
57 bool success = false;
58 juce::MemoryBlock imageData = networkFetcher.fetchBinaryBlocking(url, success);
59
60 if (success && imageData.getSize() > 0)
61 {
62 // Create image from the memory block
63 juce::MemoryInputStream inputStream(imageData, false);
64 juce::JPEGImageFormat jpegFormat;
65 juce::PNGImageFormat pngFormat;
66
67 // Try to load as JPEG first, then PNG
68 if (jpegFormat.canUnderstand(inputStream))
69 {
70 inputStream.setPosition(0);
71 image = jpegFormat.decodeImage(inputStream);
72 }
73 else
74 {
75 inputStream.setPosition(0);
76 if (pngFormat.canUnderstand(inputStream))
77 {
78 inputStream.setPosition(0);
79 image = pngFormat.decodeImage(inputStream);
80 }
81 }
82
83 // If successfully loaded image, cache it and return true
84 if (image.isValid())
85 {
86 cacheManager.saveThumbnailToCache(unitId, filename, image);
87
88 // Clear the temporary image data to free memory
89 imageData = juce::MemoryBlock();
90
91 return true;
92 }
93
94 // Clear the temporary image data even if image loading failed
95 imageData = juce::MemoryBlock();
96 }
97 }
98
99 // If we get here, loading the actual image failed, so create a placeholder
100 return createPlaceholderImage();
101}

111bool GearItem::createPlaceholderImage()

122 break;
123 case GearCategory::Preamp:
124 g.setColour(juce::Colours::red);
125 break;
126 case GearCategory::Compressor:
127 g.setColour(juce::Colours::blue);
128 break;
129 default:
130 g.setColour(juce::Colours::green);
131 break;
132 }
133
134 // Draw a rounded rectangle for the thumbnail
135 g.fillRoundedRectangle(0.0f, 0.0f, 24.0f, 24.0f, 4.0f);
136
137 // Draw the first letter of the gear name
138 g.setColour(juce::Colours::white);
139 g.setFont(16.0f);
140 g.drawText(name.substring(0, 1).toUpperCase(),
141 0, 0, 24, 24, juce::Justification::centred);
142
143 return true;
144}

214void GearItem::saveToJSON(const juce::String &filePath)
215{
216 // Create a JSON object with all of our properties
217 juce::DynamicObject::Ptr jsonObj = new juce::DynamicObject();
218
219 // Add all properties to the JSON object
220 jsonObj->setProperty("unitId", unitId);
221 jsonObj->setProperty("name", name);
222 jsonObj->setProperty("manufacturer", manufacturer);
223 jsonObj->setProperty("category", categoryString);
224 jsonObj->setProperty("version", version);
225 jsonObj->setProperty("schemaPath", schemaPath);
226 jsonObj->setProperty("thumbnailImage", thumbnailImage);
227
228 // Add instance management properties
229 jsonObj->setProperty("isInstance", isInstance);
230 jsonObj->setProperty("instanceId", instanceId);
231 jsonObj->setProperty("sourceUnitId", sourceUnitId);
232
233 // Convert GearType enum to string
234 juce::String typeStr;
235 switch (type)
236 {
237 case GearType::Series500:
238 typeStr = "500Series";
239 break;
240 case GearType::Rack19Inch:
241 typeStr = "Rack19Inch";
242 break;
243 case GearType::UserCreated:
244 typeStr = "UserCreated";
245 break;
246 case GearType::Other:
247 typeStr = "Other";
248 break;
249 }
250 jsonObj->setProperty("type", typeStr);
251
252 // Add slot size
253 jsonObj->setProperty("slotSize", slotSize);
254
255 // Add tags as an array
256 juce::Array<juce::var> tagsArray;
257 for (const auto &tag : tags)
258 tagsArray.add(tag);
259 jsonObj->setProperty("tags", tagsArray);
260
261 // Add controls as an array
262 juce::Array<juce::var> controlsArray;
263 for (const auto &control : controls)
264 {
265 juce::DynamicObject::Ptr controlObj = new juce::DynamicObject();
266
267 // Convert control type to string
268 juce::String controlTypeStr;
269 switch (control.type)
270 {
271 case GearControl::Type::Button:
272 controlTypeStr = "Button";
273 break;
274 case GearControl::Type::Fader:
275 controlTypeStr = "Fader";
276 break;
277 case GearControl::Type::Switch:
278 controlTypeStr = "Switch";
279 break;
280 }
281
282 controlObj->setProperty("type", controlTypeStr);
283 controlObj->setProperty("name", control.name);
284
285 // Convert position rectangle to JSON
286 juce::DynamicObject::Ptr posObj = new juce::DynamicObject();
287 posObj->setProperty("x", control.position.getX());
288 posObj->setProperty("y", control.position.getY());
289 posObj->setProperty("width", control.position.getWidth());
290 posObj->setProperty("height", control.position.getHeight());
291
292 controlObj->setProperty("position", posObj.get());
293 controlObj->setProperty("value", control.value);
294
295 controlsArray.add(controlObj.get());
296 }
297
298 jsonObj->setProperty("controls", controlsArray);
299
300 // Convert to JSON string
301 juce::var jsonVar(jsonObj.get());
302 juce::String jsonString = juce::JSON::toString(jsonVar);
303
304 // Write to file using the injected file system
305 fileSystem.writeFile(filePath, jsonString);
306}

319GearItem GearItem::loadFromJSON(const juce::String &filePath, INetworkFetcher &networkFetcher, IFileSystem &fileSystem)
320{
321 // Read the JSON from file using the injected file system
322 juce::String jsonString = fileSystem.readFile(filePath);
323
324 // Parse the JSON
325 juce::var jsonVar = juce::JSON::parse(jsonString);
326
327 // Check if we have a valid JSON object
328 if (!jsonVar.isObject())
329 throw std::runtime_error("Invalid JSON format in gear item file");
330
331 // Extract fields from JSON
332 juce::String unitId = jsonVar.getProperty("unitId", "");
333 juce::String name = jsonVar.getProperty("name", "");
334 juce::String manufacturer = jsonVar.getProperty("manufacturer", "");
335 juce::String categoryStr = jsonVar.getProperty("category", "");
336 juce::String version = jsonVar.getProperty("version", "1.0.0");
337 juce::String schemaPath = jsonVar.getProperty("schemaPath", "");
338 juce::String thumbnailImage = jsonVar.getProperty("thumbnailImage", "");
339
340 // Parse tags with explicit cleanup
341 juce::StringArray tags;
342 if (jsonVar.hasProperty("tags") && jsonVar["tags"].isArray())
343 {
344 juce::Array<juce::var> *tagsArray = jsonVar["tags"].getArray();
345 for (const auto &tag : *tagsArray)
346 tags.add(tag.toString());
347 // Clear the temporary array reference to release memory
348 tagsArray = nullptr;
349 }
350
351 // Determine type
352 GearType type = GearType::Other;
353 juce::String typeStr = jsonVar.getProperty("type", "");
354 if (typeStr == "500Series")
355 type = GearType::Series500;
356 else if (typeStr == "Rack19Inch")
357 type = GearType::Rack19Inch;
358 else if (typeStr == "UserCreated")
359 type = GearType::UserCreated;
360
361 // Determine category
362 GearCategory category = GearCategory::Other;
363 if (categoryStr == "equalizer" || categoryStr == "eq")
364 category = GearCategory::EQ;
365 else if (categoryStr == "compressor")
366 category = GearCategory::Compressor;
367 else if (categoryStr == "preamp")
368 category = GearCategory::Preamp;
369
370 // Get slot size
371 int slotSize = jsonVar.getProperty("slotSize", 1);
372
373 // Parse controls
374 juce::Array<GearControl> controls;
375 if (jsonVar.hasProperty("controls") && jsonVar["controls"].isArray())
376 {
377 juce::Array<juce::var> *controlsArray = jsonVar["controls"].getArray();
378 for (const auto &controlVar : *controlsArray)
379 {
380 if (!controlVar.isObject())
381 continue;
382
383 // Get control type
384 GearControl::Type controlType = GearControl::Type::Button;
385 juce::String controlTypeStr = controlVar.getProperty("type", "Button");
386 if (controlTypeStr == "Fader")
387 controlType = GearControl::Type::Fader;
388 else if (controlTypeStr == "Switch")
389 controlType = GearControl::Type::Switch;
390
391 // Get control name
392 juce::String controlName = controlVar.getProperty("name", "");
393
394 // Get position
395 juce::Rectangle<float> position;
396 if (controlVar.hasProperty("position") && controlVar["position"].isObject())
397 {
398 position.setX(controlVar["position"].getProperty("x", 0.0f));
399 position.setY(controlVar["position"].getProperty("y", 0.0f));
400 position.setWidth(controlVar["position"].getProperty("width", 0.0f));
401 position.setHeight(controlVar["position"].getProperty("height", 0.0f));
402 }
403
404 // Create control and add to array
405 GearControl control(controlType, controlName, position);
406 control.value = controlVar.getProperty("value", 0.0f);
407 controls.add(control);
408 }
409 // Clear the temporary array reference to release memory
410 controlsArray = nullptr;
411 }
412
413 // Create and return the gear item with the new constructor
414 GearItem item(unitId, name, manufacturer, categoryStr, version, schemaPath,
415 thumbnailImage, tags, networkFetcher, fileSystem, CacheManager::getDummy(), type, category, slotSize, controls);
416
417 // Image will be loaded on-demand when first accessed
418 return item;
419}
