159juce::String GearLibrary::normalizeForSearch(const juce::String &text) const
160{
161 juce::String normalized = text.toLowerCase();
162 auto ignoredChars = getIgnoredCharacters();
163
164 // Build a string of characters to remove
165 juce::String charsToRemove;
166 for (auto ignoredChar : ignoredChars)
167 {
168 charsToRemove += ignoredChar;
169 }
170
171 // Remove all ignored characters at once
172 normalized = normalized.removeCharacters(charsToRemove);
173

868 auto categoryNode = new GearTreeItem(GearTreeItem::ItemType::Category, displayName, this, &cacheManager);
869 favoritesItem->addSubItem(categoryNode);
870
871 // Sort items within this category alphabetically
872 auto &categoryItems = favoriteCategoryGroups.getReference(categoryName);
873 std::vector<GearItem *> sortedItems(categoryItems.begin(), categoryItems.end());
874 std::sort(sortedItems.begin(), sortedItems.end(),
875 [](const GearItem *a, const GearItem *b)
876 {
877 return a->name.compareIgnoreCase(b->name) < 0;
878 });
879
880 // Add gear items to this category group
881 for (auto *item : sortedItems)
882 {
883 // Find the index of this item in the original array
884 int itemIndex = -1;
885 for (int i = 0; i < gearItems.size(); ++i)
886 {
887 if (&gearItems.getReference(i) == item)
888 {
889 itemIndex = i;
890 break;
891 }
892 }
893
894 if (itemIndex >= 0)
895 {
896 categoryNode->addSubItem(new GearTreeItem(GearTreeItem::ItemType::Gear, item->name, this, &cacheManager, item, itemIndex));
897 }
898 }
899
900 // Restore the expansion state for this category
901 if (categoryExpansionState.contains(displayName))
902 {
903 // juce::Logger::writeToLog("Restoring expansion state for category: " + displayName + " = " + (categoryExpansionState[displayName] ? "open" : "closed"));
904 categoryNode->setOpenness(categoryExpansionState[displayName]);
905 }

980void GearLibrary::parseGearLibrary(const juce::String &jsonData)
981{
982 // Parse JSON data and populate gear items
983 auto json = juce::JSON::parse(jsonData);
984
985 // Check if we have a "units" array in the new format
986 if (json.hasProperty("units") && json["units"].isArray())
987 {
988 auto unitsArray = json["units"].getArray();
989 gearItems.clear();
990
991 for (auto &unitJson : *unitsArray)
992 {
993 if (unitJson.isObject())
994 {
995 auto obj = unitJson.getDynamicObject();
996
997 // Extract properties using the new format
998 juce::String unitId = obj->getProperty("unitId");
999 juce::String name = obj->getProperty("name");
1000 juce::String manufacturer = obj->getProperty("manufacturer");
1001 juce::String category = obj->getProperty("category");
1002 juce::String version = obj->getProperty("version");
1003 juce::String schemaPath = obj->getProperty("schemaPath");
1004 juce::String thumbnailImage = obj->getProperty("thumbnailImage");
1005
1006 // Process tags with explicit cleanup
1007 juce::StringArray tags;
1008 if (obj->hasProperty("tags") && obj->getProperty("tags").isArray())
1009 {
1010 auto tagsArray = obj->getProperty("tags").getArray();
1011 for (auto &tag : *tagsArray)
1012 {
1013 tags.add(tag.toString());
1014 }
1015 // Clear the temporary array reference to release memory
1016 tagsArray = nullptr;
1017 }
1018
1019 // Determine slotSize (default to 1)
1020 int slotSize = obj->hasProperty("slotSize") ? static_cast<int>(obj->getProperty("slotSize")) : 1;
1021
1022 // Create empty controls array (we'll populate this later when loading the full schema)
1023 juce::Array<GearControl> controls;
1024
1025 // Ensure schemaPath is properly formatted using our constants
1026 if (!schemaPath.startsWith("http") && !schemaPath.isEmpty())
1027 {
1028 // If it's a relative path, ensure it's relative to SCHEMAS_PATH
1029 if (!schemaPath.startsWith(RemoteResources::SCHEMAS_PATH) &&
1030 !schemaPath.startsWith("/"))
1031 {
1032 schemaPath = RemoteResources::SCHEMAS_PATH + schemaPath;
1033 }
1034 }
1035
1036 // Do the same for thumbnail images
1037 if (!thumbnailImage.startsWith("http") && !thumbnailImage.isEmpty())
1038 {
1039 // If it's a relative path and doesn't start with assets/, add the ASSETS_PATH
1040 if (!thumbnailImage.startsWith(RemoteResources::ASSETS_PATH) &&
1041 !thumbnailImage.startsWith("/"))
1042 {
1043 thumbnailImage = RemoteResources::ASSETS_PATH + thumbnailImage;
1044 }
1045 }
1046
1047 GearItem item(unitId, name, manufacturer, category, version, schemaPath,
1048 thumbnailImage, tags, networkFetcher, fileSystem, cacheManager, GearType::Other, GearCategory::Other,
1049 slotSize, controls);
1050
1051 // Add to list
1052 gearItems.add(item);
1053 }
1054 }
1055 }
1056
1057 // Update the tree view if we have a root item
1058 if (rootItem != nullptr)
1059 {
1060 rootItem->refreshSubItems();
1061
1062 // Refresh the recently used section to ensure it's populated on startup
1063 refreshRecentlyUsedSection();
1064
1065 // Refresh the favorites section to ensure it's populated on startup
1066 refreshFavoritesSection();
1067 }
1068}
