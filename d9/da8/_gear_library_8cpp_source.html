274bool GearLibrary::shouldShowItem(const GearItem &item) const
275{
276 // Check the search text
277 if (!currentSearchText.isEmpty())
278 {
279 // Normalize the search term
280 juce::String normalizedSearch = normalizeForSearch(currentSearchText);
281
282 // Debug output
283 juce::String normalizedName = normalizeForSearch(item.name);
284 juce::String normalizedManufacturer = normalizeForSearch(item.manufacturer);
285
286 // Uncomment the next line for debug output
287 juce::Logger::writeToLog("Search: '" + currentSearchText + "' -> '" + normalizedSearch + "' | Item: '" + item.name + "' -> '" + normalizedName + "' | Match: " + (normalizedName.contains(normalizedSearch) ? "YES" : "NO"));
288
289 bool matchesSearch = false;
290
291 // Check normalized name and manufacturer
292 if (normalizedName.contains(normalizedSearch) ||
293 normalizedManufacturer.contains(normalizedSearch))
294 {
295 matchesSearch = true;
296 }
297
298 // Check normalized category string
299 if (!matchesSearch && !item.categoryString.isEmpty() &&
300 normalizeForSearch(item.categoryString).contains(normalizedSearch))
301 {
302 matchesSearch = true;
303 }
304
305 // Check normalized tags
306 if (!matchesSearch && item.tags.size() > 0)
307 {
308 for (const auto &tag : item.tags)
309 {
310 if (normalizeForSearch(tag).contains(normalizedSearch))
311 {
312 matchesSearch = true;
313 break;
314 }
315 }
316 }
317
318 return matchesSearch;
319 }
320
321 // If no search text, show all items
322 return true;

804void GearLibrary::parseGearLibrary(const juce::String &jsonData)
805{
806 // Parse JSON data and populate gear items
807 auto json = juce::JSON::parse(jsonData);
808
809 // Check if we have a "units" array in the new format
810 if (json.hasProperty("units") && json["units"].isArray())
811 {
812 auto unitsArray = json["units"].getArray();
813 gearItems.clear();
814
815 for (auto &unitJson : *unitsArray)
816 {
817 if (unitJson.isObject())
818 {
819 auto obj = unitJson.getDynamicObject();
820
821 // Extract properties using the new format
822 juce::String unitId = obj->getProperty("unitId");
823 juce::String name = obj->getProperty("name");
824 juce::String manufacturer = obj->getProperty("manufacturer");
825 juce::String category = obj->getProperty("category");
826 juce::String version = obj->getProperty("version");
827 juce::String schemaPath = obj->getProperty("schemaPath");
828 juce::String thumbnailImage = obj->getProperty("thumbnailImage");
829
830 // Process tags
831 juce::StringArray tags;
832 if (obj->hasProperty("tags") && obj->getProperty("tags").isArray())
833 {
834 auto tagsArray = obj->getProperty("tags").getArray();
835 for (auto &tag : *tagsArray)
836 {
837 tags.add(tag.toString());
838 }
839 }
840
841 // Determine slotSize (default to 1)
842 int slotSize = obj->hasProperty("slotSize") ? static_cast<int>(obj->getProperty("slotSize")) : 1;
843
844 // Create empty controls array (we'll populate this later when loading the full schema)
845 juce::Array<GearControl> controls;
846
847 // Ensure schemaPath is properly formatted using our constants
848 if (!schemaPath.startsWith("http") && !schemaPath.isEmpty())
849 {
850 // If it's a relative path, ensure it's relative to SCHEMAS_PATH
851 if (!schemaPath.startsWith(RemoteResources::SCHEMAS_PATH) &&
852 !schemaPath.startsWith("/"))
853 {
854 schemaPath = RemoteResources::SCHEMAS_PATH + schemaPath;
855 }
856 }
857
858 // Do the same for thumbnail images
859 if (!thumbnailImage.startsWith("http") && !thumbnailImage.isEmpty())
860 {
861 // If it's a relative path and doesn't start with assets/, add the ASSETS_PATH
862 if (!thumbnailImage.startsWith(RemoteResources::ASSETS_PATH) &&
863 !thumbnailImage.startsWith("/"))
864 {
865 thumbnailImage = RemoteResources::ASSETS_PATH + thumbnailImage;
866 }
867 }
868
869 GearItem item(unitId, name, manufacturer, category, version, schemaPath,
870 thumbnailImage, tags, fetcher, GearType::Other, GearCategory::Other,
871 slotSize, controls);
872
873 // If thumbnailImage is provided, try to load the image
874 if (thumbnailImage.isNotEmpty())
875 {
876 // For now, we'll just use our placeholder image code in loadImage()
877 item.loadImage();
878 }
879
880 // Add to list
881 gearItems.add(item);
882 }
883 }
884 }
885 // Fallback to the old "gear" array format for backwards compatibility
886 else if (json.hasProperty("gear") && json["gear"].isArray())
887 {
888 auto gearArray = json["gear"].getArray();
889 gearItems.clear();
890
891 for (auto &gearJson : *gearArray)
892 {
893 if (gearJson.isObject())
894 {
895 auto obj = gearJson.getDynamicObject();
896
897 // Extract properties using the old format
898 juce::String name = obj->getProperty("name");
899 juce::String manufacturer = obj->getProperty("manufacturer");
900
901 // Determine type
902 GearType type;
903 juce::String typeStr = obj->getProperty("type");
904 if (typeStr == "500Series")
905 type = GearType::Series500;
906 else if (typeStr == "Rack19Inch")
907 type = GearType::Rack19Inch;
908 else
909 type = GearType::UserCreated;
910
911 // Determine category
912 GearCategory category;
913 juce::String catStr = obj->getProperty("category");
914 if (catStr == "EQ")
915 category = GearCategory::EQ;
916 else if (catStr == "Compressor")
917 category = GearCategory::Compressor;
918 else if (catStr == "Preamp")
919 category = GearCategory::Preamp;
920 else
921 category = GearCategory::Other;
922
923 int slotSize = obj->getProperty("slotSize");
924 juce::String thumbnailUrl = obj->getProperty("thumbnailUrl");
925
926 // Create an empty array of controls
927 juce::Array<GearControl> controls;
928
929 // Create gear item using the legacy constructor
930 GearItem item(name, manufacturer, type, category, slotSize, thumbnailUrl, controls, fetcher);
931
932 // If thumbnailUrl is provided, try to load the image
933 if (thumbnailUrl.isNotEmpty())
934 {
935 item.loadImage();
936 }
937
938 // Add to list
939 gearItems.add(item);
940 }
941 }
942 }
943
944 // Update the tree view if we have a root item
945 if (rootItem != nullptr)
946 rootItem->refreshSubItems();
947}
