244juce::String GearLibrary::normalizeForSearch(const juce::String &text) const
245{

251 for (auto ignoredChar : ignoredChars)

271bool GearLibrary::shouldShowItem(const GearItem &item) const

273 // Check the search text

277 juce::String normalizedSearch = normalizeForSearch(currentSearchText);
278
279 // Debug output
280 juce::String normalizedName = normalizeForSearch(item.name);
281 juce::String normalizedManufacturer = normalizeForSearch(item.manufacturer);

283 // Uncomment the next line for debug output
284 // juce::Logger::writeToLog("Search: '" + currentSearchText + "' -> '" + normalizedSearch + "' | Item: '" + item.name + "' -> '" + normalizedName + "' | Match: " + (normalizedName.contains(normalizedSearch) ? "YES" : "NO"));

288 // Check normalized name and manufacturer
289 if (normalizedName.contains(normalizedSearch) ||
290 normalizedManufacturer.contains(normalizedSearch))
291 {
292 matchesSearch = true;
293 }
294
295 // Check normalized category string
296 if (!matchesSearch && !item.categoryString.isEmpty() &&
297 normalizeForSearch(item.categoryString).contains(normalizedSearch))
298 {
299 matchesSearch = true;
300 }

303 if (!matchesSearch && item.tags.size() > 0)

307 if (normalizeForSearch(tag).contains(normalizedSearch))
308 {
309 matchesSearch = true;
310 break;
311 }
312 }
313 }
314
315 return matchesSearch;
316 }

318 // If no search text, show all items

900 juce::HashMap<juce::String, juce::Array<GearItem *>> favoriteCategoryGroups;
901
902 for (const auto &unitId : favorites)
903 {
904 // Find the gear item in the library
905 for (int i = 0; i < gearItems.size(); ++i)
906 {
907 const auto &item = gearItems.getReference(i);
908 if (item.unitId == unitId)
909 {
910 // Get the category string or derive it from the enum
911 juce::String category;
912 if (!item.categoryString.isEmpty())
913 {
914 category = item.categoryString.toLowerCase();
915 }
916 else
917 {
918 // Fallback to enum if string is empty
919 switch (item.category)
920 {
921 case GearCategory::EQ:
922 category = "equalizer";
923 break;
924 case GearCategory::Compressor:
925 category = "compressor";
926 break;
927 case GearCategory::Preamp:
928 category = "preamp";
929 break;
930 case GearCategory::Other:
931 category = "other";
932 break;
933 }
934 }
935
936 if (!favoriteCategoryGroups.contains(category))
937 favoriteCategoryGroups.set(category, juce::Array<GearItem *>());
938
939 favoriteCategoryGroups.getReference(category).add(const_cast<GearItem *>(&item));
940 // juce::Logger::writeToLog(" - Added item to existing section: " + item.name);
941 break;
942 }
943 }
944 }
945
946 // Sort categories alphabetically
947 juce::StringArray sortedCategoryNames;
948 for (auto it = favoriteCategoryGroups.begin(); it != favoriteCategoryGroups.end(); ++it)
949 {
950 sortedCategoryNames.add(it.getKey());
951 }
952 sortedCategoryNames.sort(true); // true = ignore case
953
954 // Add category groups as sub-items and restore their expansion state
955 for (const auto &categoryName : sortedCategoryNames)
956 {
957 juce::String displayName = categoryName.substring(0, 1).toUpperCase() + categoryName.substring(1);
958
959 auto categoryNode = new GearTreeItem(GearTreeItem::ItemType::Category, displayName, this, &cacheManager);
960 favoritesItem->addSubItem(categoryNode);
961
962 // Sort items within this category alphabetically
963 auto &categoryItems = favoriteCategoryGroups.getReference(categoryName);
964 std::vector<GearItem *> sortedItems(categoryItems.begin(), categoryItems.end());
965 std::sort(sortedItems.begin(), sortedItems.end(),
966 [](const GearItem *a, const GearItem *b)
967 {
968 return a->name.compareIgnoreCase(b->name) < 0;
969 });
970
971 // Add gear items to this category group
972 for (auto *item : sortedItems)
973 {
974 // Find the index of this item in the original array
975 int itemIndex = -1;
976 for (int i = 0; i < gearItems.size(); ++i)
977 {
978 if (&gearItems.getReference(i) == item)
979 {
980 itemIndex = i;
981 break;
982 }
983 }
984
985 if (itemIndex >= 0)
986 {
987 categoryNode->addSubItem(new GearTreeItem(GearTreeItem::ItemType::Gear, item->name, this, &cacheManager, item, itemIndex));
988 }
989 }
990
991 // Restore the expansion state for this category
992 if (categoryExpansionState.contains(displayName))
993 {
994 // juce::Logger::writeToLog("Restoring expansion state for category: " + displayName + " = " + (categoryExpansionState[displayName] ? "open" : "closed"));
995 categoryNode->setOpenness(categoryExpansionState[displayName]);
996 }
997 }
998 }
999
1000 gearTreeView->repaint();
1001 }

1142void GearLibrary::parseGearLibrary(const juce::String &jsonData)
1143{
1144 // Parse JSON data and populate gear items
1145 auto json = juce::JSON::parse(jsonData);
1146
1147 // Check if we have a "units" array in the new format
1148 if (json.hasProperty("units") && json["units"].isArray())
1149 {
1150 auto unitsArray = json["units"].getArray();
1151 gearItems.clear();
1152
1153 for (auto &unitJson : *unitsArray)
1154 {
1155 if (unitJson.isObject())
1156 {
1157 auto obj = unitJson.getDynamicObject();
1158
1159 // Extract properties using the new format
1160 juce::String unitId = obj->getProperty("unitId");
1161 juce::String name = obj->getProperty("name");
1162 juce::String manufacturer = obj->getProperty("manufacturer");
1163 juce::String category = obj->getProperty("category");
1164 juce::String version = obj->getProperty("version");
1165 juce::String schemaPath = obj->getProperty("schemaPath");
1166 juce::String thumbnailImage = obj->getProperty("thumbnailImage");
1167
1168 // Process tags
1169 juce::StringArray tags;
1170 if (obj->hasProperty("tags") && obj->getProperty("tags").isArray())
1171 {
1172 auto tagsArray = obj->getProperty("tags").getArray();
1173 for (auto &tag : *tagsArray)
1174 {
1175 tags.add(tag.toString());
1176 }
1177 }
1178
1179 // Determine slotSize (default to 1)
1180 int slotSize = obj->hasProperty("slotSize") ? static_cast<int>(obj->getProperty("slotSize")) : 1;
1181
1182 // Create empty controls array (we'll populate this later when loading the full schema)
1183 juce::Array<GearControl> controls;
1184
1185 // Ensure schemaPath is properly formatted using our constants
1186 if (!schemaPath.startsWith("http") && !schemaPath.isEmpty())
1187 {
1188 // If it's a relative path, ensure it's relative to SCHEMAS_PATH
1189 if (!schemaPath.startsWith(RemoteResources::SCHEMAS_PATH) &&
1190 !schemaPath.startsWith("/"))
1191 {
1192 schemaPath = RemoteResources::SCHEMAS_PATH + schemaPath;
1193 }
1194 }
1195
1196 // Do the same for thumbnail images
1197 if (!thumbnailImage.startsWith("http") && !thumbnailImage.isEmpty())
1198 {
1199 // If it's a relative path and doesn't start with assets/, add the ASSETS_PATH
1200 if (!thumbnailImage.startsWith(RemoteResources::ASSETS_PATH) &&
1201 !thumbnailImage.startsWith("/"))
1202 {
1203 thumbnailImage = RemoteResources::ASSETS_PATH + thumbnailImage;
1204 }
1205 }
1206
1207 GearItem item(unitId, name, manufacturer, category, version, schemaPath,
1208 thumbnailImage, tags, networkFetcher, fileSystem, cacheManager, GearType::Other, GearCategory::Other,
1209 slotSize, controls);
1210
1211 // If thumbnailImage is provided, try to load the image
1212 if (thumbnailImage.isNotEmpty())
1213 {
1214 // For now, we'll just use our placeholder image code in loadImage()
1215 item.loadImage();
1216 }
1217
1218 // Add to list
1219 gearItems.add(item);
1220 }
1221 }
1222 }
1223 // Fallback to the old "gear" array format for backwards compatibility
1224 else if (json.hasProperty("gear") && json["gear"].isArray())
1225 {
1226 auto gearArray = json["gear"].getArray();
1227 gearItems.clear();
1228
1229 for (auto &gearJson : *gearArray)
1230 {
1231 if (gearJson.isObject())
1232 {
1233 auto obj = gearJson.getDynamicObject();
1234
1235 // Extract properties using the old format
1236 juce::String name = obj->getProperty("name");
1237 juce::String manufacturer = obj->getProperty("manufacturer");
1238
1239 // Determine type
1240 GearType type;
1241 juce::String typeStr = obj->getProperty("type");
1242 if (typeStr == "500Series")
1243 type = GearType::Series500;
1244 else if (typeStr == "Rack19Inch")
1245 type = GearType::Rack19Inch;
1246 else
1247 type = GearType::UserCreated;
1248
1249 // Determine category
1250 GearCategory category;
1251 juce::String catStr = obj->getProperty("category");
1252 if (catStr == "EQ")
1253 category = GearCategory::EQ;
1254 else if (catStr == "Compressor")
1255 category = GearCategory::Compressor;
1256 else if (catStr == "Preamp")
1257 category = GearCategory::Preamp;
1258 else
1259 category = GearCategory::Other;
1260
1261 int slotSize = obj->getProperty("slotSize");
1262 juce::String thumbnailUrl = obj->getProperty("thumbnailUrl");
1263
1264 // Create an empty array of controls
1265 juce::Array<GearControl> controls;
1266
1267 // Create gear item using the legacy constructor
1268 GearItem item(name, manufacturer, type, category, slotSize, thumbnailUrl, controls, networkFetcher, fileSystem, cacheManager);
1269
1270 // If thumbnailUrl is provided, try to load the image
1271 if (thumbnailUrl.isNotEmpty())
1272 {
1273 item.loadImage();
1274 }
1275
1276 // Add to list
1277 gearItems.add(item);
1278 }
1279 }
1280 }
1281
1282 // Update the tree view if we have a root item
1283 if (rootItem != nullptr)
1284 {
1285 rootItem->refreshSubItems();
1286
1287 // Refresh the recently used section to ensure it's populated on startup
1288 refreshRecentlyUsedSection();
1289
1290 // Refresh the favorites section to ensure it's populated on startup
1291 refreshFavoritesSection();
