244bool GearLibrary::shouldShowItem(const GearItem &item) const
245{
246 // First check the search text
247 if (!currentSearchText.isEmpty())
248 {
249 bool matchesSearch = false;
250
251 // Check name and manufacturer
252 if (item.name.containsIgnoreCase(currentSearchText) ||
253 item.manufacturer.containsIgnoreCase(currentSearchText))
254 {
255 matchesSearch = true;
256 }
257
258 // Check category string
259 if (!matchesSearch && !item.categoryString.isEmpty() &&
260 item.categoryString.containsIgnoreCase(currentSearchText))
261 {
262 matchesSearch = true;
263 }
264
265 // Check tags
266 if (!matchesSearch && item.tags.size() > 0)
267 {
268 for (const auto &tag : item.tags)
269 {
270 if (tag.containsIgnoreCase(currentSearchText))
271 {
272 matchesSearch = true;
273 break;
274 }
275 }
276 }
277
278 if (!matchesSearch)
279 return false;
280 }
281
282 // Then check the filter
283 if (currentFilter.first == FilterCategory::All)
284 return true;
285
286 if (currentFilter.first == FilterCategory::Type)
287 {
288 if (currentFilter.second == "500Series")
289 return item.type == GearType::Series500;
290 if (currentFilter.second == "Rack19Inch")
291 return item.type == GearType::Rack19Inch;
292 if (currentFilter.second == "UserCreated")
293 return item.type == GearType::UserCreated;
294 }
295 else if (currentFilter.first == FilterCategory::Category)
296 {
297 // Check both the enum category and the string category for flexibility
298 if (currentFilter.second == "EQ" || currentFilter.second == "equalizer")
299 return (item.category == GearCategory::EQ ||
300 item.categoryString.equalsIgnoreCase("equalizer") ||
301 item.categoryString.equalsIgnoreCase("eq"));
302
303 if (currentFilter.second == "Preamp" || currentFilter.second == "preamp")
304 return (item.category == GearCategory::Preamp ||
305 item.categoryString.equalsIgnoreCase("preamp"));
306
307 if (currentFilter.second == "Compressor" || currentFilter.second == "compressor")
308 return (item.category == GearCategory::Compressor ||
309 item.categoryString.equalsIgnoreCase("compressor"));
310
311 if (currentFilter.second == "Other" || currentFilter.second == "other")
312 return (item.category == GearCategory::Other ||
313 item.categoryString.equalsIgnoreCase("other"));
314 }
315
316 return true;
317}

324void GearLibrary::updateFilteredItems()
325{
326 // First update the legacy list box
327 if (gearListBox)
328 {
329 gearListBox->updateContent();
330 gearListBox->repaint();
331 }
332
333 // Then update the tree view
334 if (rootItem && gearTreeView)
335 {
336 bool isSearching = !currentSearchText.isEmpty();
337
338 // If we're searching or the root item has no children yet, refresh from scratch
339 if (isSearching || rootItem->getNumSubItems() == 0)
340 {
341 rootItem->clearSubItems();
342 rootItem->refreshSubItems();
343 }
344
345 // Recursively expand and hide/show nodes based on search results
346 if (isSearching)
347 {
348 // First see if any item in the entire tree matches the search
349 bool anyMatches = false;
350 for (int i = 0; i < gearItems.size(); ++i)
351 {
352 if (shouldShowItem(gearItems.getReference(i)))
353 {
354 anyMatches = true;
355 break;
356 }
357 }
358
359 // If we have matches, expand everything to show them
360 if (anyMatches)
361 {
362 // Make sure the root categories node is expanded
363 for (int i = 0; i < rootItem->getNumSubItems(); ++i)
364 {
365 if (auto categoriesNode = dynamic_cast<GearTreeItem *>(rootItem->getSubItem(i)))
366 {
367 categoriesNode->setOpen(true);
368
369 // For each category, check if it contains matching items
370 for (int j = 0; j < categoriesNode->getNumSubItems(); ++j)
371 {
372 if (auto categoryNode = dynamic_cast<GearTreeItem *>(categoriesNode->getSubItem(j)))
373 {
374 bool categoryHasMatches = false;
375
376 // If the node hasn't loaded its items yet, refresh it
377 if (categoryNode->getNumSubItems() == 0)
378 {
379 categoryNode->refreshSubItems();
380 }
381
382 // Check each gear item to see if it matches
383 for (int k = 0; k < categoryNode->getNumSubItems(); ++k)
384 {
385 if (auto gearNode = dynamic_cast<GearTreeItem *>(categoryNode->getSubItem(k)))
386 {
387 if (GearItem *gearItem = gearNode->getGearItem())
388 {
389 // Check if this item matches the search
390 bool matches = shouldShowItem(*gearItem);
391 gearNode->setVisible(matches);
392
393 if (matches)
394 categoryHasMatches = true;
395 }
396 }
397 }
398
399 // Only show and expand categories that have matching items
400 categoryNode->setVisible(categoryHasMatches);
401 categoryNode->setOpen(categoryHasMatches);
402 }
403 }
404 }
405 }
406 }
407 else
408 {
409 // If no matches, don't expand anything
410 rootItem->setOpenness(false);
411 }
412 }
413 else
414 {
415 // When not searching, make all nodes visible
416 for (int i = 0; i < rootItem->getNumSubItems(); ++i)
417 {
418 if (auto categoriesNode = dynamic_cast<GearTreeItem *>(rootItem->getSubItem(i)))
419 {
420 categoriesNode->setVisible(true);
421
422 for (int j = 0; j < categoriesNode->getNumSubItems(); ++j)
423 {
424 if (auto categoryNode = dynamic_cast<GearTreeItem *>(categoriesNode->getSubItem(j)))
425 {
426 categoryNode->setVisible(true);
427
428 for (int k = 0; k < categoryNode->getNumSubItems(); ++k)
429 {
430 if (auto gearNode = dynamic_cast<GearTreeItem *>(categoryNode->getSubItem(k)))
431 {
432 gearNode->setVisible(true);
433 }
434 }
435 }
436 }
437 }
438 }
439 }
440
441 gearTreeView->repaint();
442 }

605void GearLibrary::parseGearLibrary(const juce::String &jsonData)
606{
607 // Parse JSON data and populate gear items
608 auto json = juce::JSON::parse(jsonData);
609
610 // Check if we have a "units" array in the new format
611 if (json.hasProperty("units") && json["units"].isArray())
612 {
613 auto unitsArray = json["units"].getArray();
614 gearItems.clear();
615
616 for (auto &unitJson : *unitsArray)
617 {
618 if (unitJson.isObject())
619 {
620 auto obj = unitJson.getDynamicObject();
621
622 // Extract properties using the new format
623 juce::String unitId = obj->getProperty("unitId");
624 juce::String name = obj->getProperty("name");
625 juce::String manufacturer = obj->getProperty("manufacturer");
626 juce::String category = obj->getProperty("category");
627 juce::String version = obj->getProperty("version");
628 juce::String schemaPath = obj->getProperty("schemaPath");
629 juce::String thumbnailImage = obj->getProperty("thumbnailImage");
630
631 // Process tags
632 juce::StringArray tags;
633 if (obj->hasProperty("tags") && obj->getProperty("tags").isArray())
634 {
635 auto tagsArray = obj->getProperty("tags").getArray();
636 for (auto &tag : *tagsArray)
637 {
638 tags.add(tag.toString());
639 }
640 }
641
642 // Determine slotSize (default to 1)
643 int slotSize = obj->hasProperty("slotSize") ? static_cast<int>(obj->getProperty("slotSize")) : 1;
644
645 // Create empty controls array (we'll populate this later when loading the full schema)
646 juce::Array<GearControl> controls;
647
648 // Ensure schemaPath is properly formatted using our constants
649 if (!schemaPath.startsWith("http") && !schemaPath.isEmpty())
650 {
651 // If it's a relative path, ensure it's relative to SCHEMAS_PATH
652 if (!schemaPath.startsWith(RemoteResources::SCHEMAS_PATH) &&
653 !schemaPath.startsWith("/"))
654 {
655 schemaPath = RemoteResources::SCHEMAS_PATH + schemaPath;
656 }
657 }
658
659 // Do the same for thumbnail images
660 if (!thumbnailImage.startsWith("http") && !thumbnailImage.isEmpty())
661 {
662 // If it's a relative path and doesn't start with assets/, add the ASSETS_PATH
663 if (!thumbnailImage.startsWith(RemoteResources::ASSETS_PATH) &&
664 !thumbnailImage.startsWith("/"))
665 {
666 thumbnailImage = RemoteResources::ASSETS_PATH + thumbnailImage;
667 }
668 }
669
670 GearItem item(unitId, name, manufacturer, category, version, schemaPath,
671 thumbnailImage, tags, fetcher, GearType::Other, GearCategory::Other,
672 slotSize, controls);
673
674 // If thumbnailImage is provided, try to load the image
675 if (thumbnailImage.isNotEmpty())
676 {
677 // For now, we'll just use our placeholder image code in loadImage()
678 item.loadImage();
679 }
680
681 // Add to list
682 gearItems.add(item);
683 }
684 }
685 }
686 // Fallback to the old "gear" array format for backwards compatibility
687 else if (json.hasProperty("gear") && json["gear"].isArray())
688 {
689 auto gearArray = json["gear"].getArray();
690 gearItems.clear();
691
692 for (auto &gearJson : *gearArray)
693 {
694 if (gearJson.isObject())
695 {
696 auto obj = gearJson.getDynamicObject();
697
698 // Extract properties using the old format
699 juce::String name = obj->getProperty("name");
700 juce::String manufacturer = obj->getProperty("manufacturer");
701
702 // Determine type
703 GearType type;
704 juce::String typeStr = obj->getProperty("type");
705 if (typeStr == "500Series")
706 type = GearType::Series500;
707 else if (typeStr == "Rack19Inch")
708 type = GearType::Rack19Inch;
709 else
710 type = GearType::UserCreated;
711
712 // Determine category
713 GearCategory category;
714 juce::String catStr = obj->getProperty("category");
715 if (catStr == "EQ")
716 category = GearCategory::EQ;
717 else if (catStr == "Compressor")
718 category = GearCategory::Compressor;
719 else if (catStr == "Preamp")
720 category = GearCategory::Preamp;
721 else
722 category = GearCategory::Other;
723
724 int slotSize = obj->getProperty("slotSize");
725 juce::String thumbnailUrl = obj->getProperty("thumbnailUrl");
726
727 // Create an empty array of controls
728 juce::Array<GearControl> controls;
729
730 // Create gear item using the legacy constructor
731 GearItem item(name, manufacturer, type, category, slotSize, thumbnailUrl, controls, fetcher);
732
733 // If thumbnailUrl is provided, try to load the image
734 if (thumbnailUrl.isNotEmpty())
735 {
736 item.loadImage();
737 }
738
739 // Add to list
740 gearItems.add(item);
741 }
742 }
743 }
744
745 // Update the tree view if we have a root item
746 if (rootItem != nullptr)
747 rootItem->refreshSubItems();
748}

869void GearLibrary::parseFilterOptions(const juce::String &jsonData)
870{
871 // This method is retained for backwards compatibility but simplified
872 // since we no longer use the filter box
873 auto json = juce::JSON::parse(jsonData);
874
875 if (json.hasProperty("filters") && json["filters"].isArray())
876 {
877 auto filterArray = json["filters"].getArray();
878 filterOptions.clear();
879
880 // Always add "All Items" as the first option
881 filterOptions.add({"All Items", FilterCategory::All, ""});
882
883 for (auto &filterJson : *filterArray)
884 {
885 if (filterJson.isObject())
886 {
887 auto obj = filterJson.getDynamicObject();
888
889 juce::String displayName = obj->getProperty("displayName");
890 juce::String category = obj->getProperty("category");
891 juce::String value = obj->getProperty("value");
892
893 FilterCategory filterCategory;
894 if (category == "Type")
895 filterCategory = FilterCategory::Type;
896 else if (category == "Category")
897 filterCategory = FilterCategory::Category;
898 else
899 continue; // Skip invalid categories
900
901 filterOptions.add({displayName, filterCategory, value});
902 }
903 }
904 }
