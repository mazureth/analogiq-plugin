94int Rack::getSlotHeight(int slotIndex) const
95{
96 if (slotIndex < 0 || slotIndex >= slots.size())
97 return getDefaultSlotHeight();
98
99 RackSlot *slot = slots[slotIndex];
100 if (slot == nullptr || slot->isAvailable())
101 return getDefaultSlotHeight();
102
103 // If the slot has a gear item with a faceplate image, use the image's height plus padding
104 GearItem *item = slot->getGearItem();
105 if (item != nullptr && item->faceplateImage.isValid())
106 {
107 // Calculate a reasonable height based on the faceplate image
108 // Use aspect ratio of the image, but constrained to reasonable bounds
109 int imageHeight = item->faceplateImage.getHeight();
110 int imageWidth = item->faceplateImage.getWidth();
111
112 if (imageHeight > 0 && imageWidth > 0)
113 {
114 // Calculate what the height would be if the width matched the slot width
115 // Add extra padding for controls and slot UI elements
116 int effectiveSlotWidth = rackContainer->getWidth() - (2 * slotSpacing);
117 int scaledHeight = (imageHeight * effectiveSlotWidth) / imageWidth;
118
119 // Add padding for slot UI elements (buttons, labels, etc.)
120 int paddedHeight = scaledHeight + 40; // 20px padding top and bottom
121
122 // Constrain to reasonable bounds
123 return juce::jlimit(100, 400, paddedHeight);
124 }
125 }
126
127 // Default height if no special considerations apply
128 return getDefaultSlotHeight();
129}

137void Rack::resized()
138{
139 // Size the viewport to fill our area
140 auto area = getLocalBounds();
141 rackViewport->setBounds(area);
142
143 // Get the available width from the viewport
144 const int availableWidth = rackViewport->getWidth();
145
146 // Calculate total height needed for all slots with their variable heights
147 int totalHeight = slotSpacing; // Start with top spacing
148 for (int i = 0; i < slots.size(); ++i)
149 {
150 totalHeight += getSlotHeight(i) + slotSpacing;
151 }
152
153 // Size the container to fit all slots with spacing
154 const int containerWidth = availableWidth; // Use full viewport width
155 rackContainer->setSize(containerWidth, totalHeight);
156
157 // Calculate the slot width based on container width minus margins
158 const int effectiveSlotWidth = containerWidth - (2 * slotSpacing);
159
160 // Position the slots within the container
161 int currentY = slotSpacing;
162 for (int i = 0; i < slots.size(); ++i)
163 {
164 auto *slot = slots[i];
165 int slotHeight = getSlotHeight(i);
166
167 slot->setBounds(
168 slotSpacing,
169 currentY,
170 effectiveSlotWidth,
171 slotHeight);
172
173 currentY += slotHeight + slotSpacing;
174 }

259{
260 // Find the nearest slot to the drop position
261 RackSlot *targetSlot = findNearestSlot(details.localPosition);
262 if (targetSlot == nullptr)
263 {
264 return;
265 }
266
267 // Clear highlights
268 for (auto *slot : slots)
269 {
270 slot->setHighlighted(false);
271 }
272
273 // Handle drops from GearLibrary (legacy list box)
274 if (details.description.isInt() && gearLibrary != nullptr)
275 {
276 int gearIndex = static_cast<int>(details.description);
277 GearItem *item = gearLibrary->getGearItem(gearIndex);
278
279 if (item != nullptr)
280 {
281 // Create a new instance of the item
282 GearItem *newItem = new GearItem(*item, networkFetcher); // Use copy constructor
283 targetSlot->setGearItem(newItem);
284
285 // Fetch schema for this item
286 fetchSchemaForGearItem(newItem);

288 }
289 // Handle drops from TreeView
290 else if (details.description.isString() && gearLibrary != nullptr)
291 {
292 juce::String desc = details.description.toString();
293 if (desc.startsWith("GEAR:"))
294 {
295 // Parse the gear index from the drag descriptor
296 juce::StringArray parts;
297 parts.addTokens(desc, ":", "");
298
299 if (parts.size() >= 3)
300 {
301 int gearIndex = parts[1].getIntValue();
302 GearItem *item = gearLibrary->getGearItem(gearIndex);
303
304 if (item != nullptr)
305 {
306 // Create a new instance of the item
307 GearItem *newItem = new GearItem(*item, networkFetcher); // Use copy constructor
308 targetSlot->setGearItem(newItem);
309
310 // Fetch schema for this item
311 fetchSchemaForGearItem(newItem);
312 }
313 }
314 }
315 }
316 // Handle drops from other RackSlots (rearranging)
317 else if (auto *sourceSlot = dynamic_cast<RackSlot *>(details.sourceComponent.get()))
318 {
319 if (sourceSlot != targetSlot && sourceSlot->getGearItem() != nullptr)
320 {
321 // Swap the items between slots
322 GearItem *sourceItem = sourceSlot->getGearItem();

337void Rack::rearrangeGearAsSortableList(int sourceSlotIndex, int targetSlotIndex)
338{
339 // Validate indices
340 if (sourceSlotIndex < 0 || sourceSlotIndex >= slots.size() ||
341 targetSlotIndex < 0 || targetSlotIndex >= slots.size() ||
342 sourceSlotIndex == targetSlotIndex)
343 {
344 return;
345 }
346
347 // Get pointers to the source and target slots
348 RackSlot *sourceSlot = slots[sourceSlotIndex];
349 RackSlot *targetSlot = slots[targetSlotIndex];
350
351 if (sourceSlot == nullptr || targetSlot == nullptr)
352 {
353 return;
354 }
355
356 // Get the gear items from both slots
357 GearItem *sourceGearItem = sourceSlot->getGearItem();
358 GearItem *targetGearItem = targetSlot->getGearItem();
359
360 if (sourceGearItem == nullptr)
361 {
362 return;
363 }
364
365 // Simply swap the two slots
366
367 // First clear both slots
368 sourceSlot->clearGearItem();
369 targetSlot->clearGearItem();
370
371 // Then set the items in their new positions
372 targetSlot->setGearItem(sourceGearItem);
373
374 // If the target slot had an item, move it to the source slot
375 if (targetGearItem != nullptr)
376 {
377 sourceSlot->setGearItem(targetGearItem);
378 }
379
380 // Update the rack view - call resized() on the rack itself to recalculate all slot heights
381 resized();
382}

390RackSlot *Rack::findNearestSlot(const juce::Point<int> &position)
391{
392 // This method is kept for compatibility with drops from GearLibrary
393 // but we've simplified it since we're not using it for reordering
394
395 // Convert global position to container coordinates
396 auto containerPos = rackContainer->getLocalPoint(this, position);
397
398 // First check for a direct hit
399 for (auto *slot : slots)
400 {
401 if (slot->getBounds().contains(containerPos))
402 {
403 return slot;
404 }
405 }
406
407 // If no direct hit, find the closest slot
408 RackSlot *bestSlot = nullptr;
409 int bestDistance = std::numeric_limits<int>::max();
410
411 for (auto *slot : slots)
412 {
413 juce::Rectangle<int> slotBounds = slot->getBounds();
414 int distance = slotBounds.getCentre().getDistanceFrom(containerPos);
415
416 if (distance < bestDistance)
417 {
418 bestDistance = distance;
419 bestSlot = slot;
420 }
421 }
422
423 return bestSlot;
424}

525void Rack::parseSchema(const juce::String &schemaData, GearItem *item)
526{
527 // Parse the JSON schema
528 auto schemaJson = juce::JSON::parse(schemaData);
529 if (!schemaJson.isObject())
530 {
531 return;
532 }
533
534 // Look for faceplate image properties
535 juce::StringArray faceplateProperties = {"faceplateImage", "thumbnailImage"};
536 juce::String faceplateImagePath;
537 bool foundFaceplate = false;
538
539 for (const auto &propName : faceplateProperties)
540 {
541 if (schemaJson.hasProperty(propName))
542 {
543 faceplateImagePath = schemaJson.getProperty(propName, "").toString();
544 if (faceplateImagePath.isNotEmpty())
545 {
546 foundFaceplate = true;
547
548 // Store the path in the gear item
549 item->faceplateImagePath = faceplateImagePath;
550
551 // Fetch the faceplate image
552 fetchFaceplateImage(item);
553 break;
554 }
555 }
556 }
557
558 if (!foundFaceplate)
559 {
560 }
561
562 // Update controls if available
563 if (schemaJson.hasProperty("controls") && schemaJson["controls"].isArray())
564 {
565 item->controls.clear();
566
567 auto controlsArray = schemaJson["controls"].getArray();
568 for (auto &controlVar : *controlsArray)
569 {
570 if (!controlVar.isObject())
571 continue;
572
573 // Get control type
574 GearControl::Type controlType = GearControl::Type::Button;
575 juce::String controlTypeStr = controlVar.getProperty("type", "button").toString().toLowerCase();
576 if (controlTypeStr == "fader")
577 controlType = GearControl::Type::Fader;
578 else if (controlTypeStr == "switch")
579 controlType = GearControl::Type::Switch;
580 else if (controlTypeStr == "knob")
581 controlType = GearControl::Type::Knob;
582
583 // Get control name and ID
584 juce::String controlName = controlVar.getProperty("label", "");
585 juce::String controlId = controlVar.getProperty("id", ""); // Get ID from schema
586
587 // If no ID is provided, generate one from the label
588 if (controlId.isEmpty())
589 {
590 controlId = controlName.toLowerCase().replaceCharacters(" ", "-");
591 }
592
593 // Check if we already have a control with this ID
594 bool controlExists = false;
595 for (const auto &existingControl : item->controls)
596 {
597 if (existingControl.id == controlId)
598 {
599 controlExists = true;
600 break;
601 }
602 }
603
604 if (controlExists)
605 continue;
606
607 // Get position
608 juce::Rectangle<float> position;
609 if (controlVar.hasProperty("position") && controlVar["position"].isObject())
610 {
611 position.setX(controlVar["position"].getProperty("x", 0.0f));
612 position.setY(controlVar["position"].getProperty("y", 0.0f));
613 position.setWidth(controlVar["position"].getProperty("width", 0.0f));
614 position.setHeight(controlVar["position"].getProperty("height", 0.0f));
615 }
616
617 // Create control
618 GearControl control(controlType, controlName, position);
619 control.id = controlId; // Set the control ID
620 control.value = controlVar.getProperty("value", 0.0f);
621
622 // Set type-specific properties
623 switch (control.type)
624 {
625 case GearControl::Type::Switch:
626 {
627 // Get switch-specific properties
628 control.orientation = controlVar.getProperty("orientation", "vertical").toString();
629 control.currentIndex = controlVar.getProperty("currentIndex", 0);
630 control.value = (float)control.currentIndex; // Set value based on currentIndex
631 control.initialValue = control.value; // Store initial value
632 control.image = controlVar.getProperty("image", "").toString();
633
634 // Parse options array
635 if (controlVar.hasProperty("options") && controlVar["options"].isArray())
636 {
637 auto optionsArray = controlVar["options"].getArray();
638 for (const auto &option : *optionsArray)
639 {
640 if (option.isObject())
641 {
642 GearControl::SwitchOptionFrame frame;
643 frame.value = option.getProperty("value", "").toString();
644 frame.label = option.getProperty("label", "").toString();
645
646 // Parse frame data
647 if (option.hasProperty("frame") && option["frame"].isObject())
648 {
649 auto frameObj = option["frame"];
650 frame.x = frameObj.getProperty("x", 0);
651 frame.y = frameObj.getProperty("y", 0);
652 frame.width = frameObj.getProperty("width", 0);
653 frame.height = frameObj.getProperty("height", 0);
654 }
655
656 control.switchFrames.add(frame);
657 control.options.add(frame.value);
658 }
659 }
660 }
661
662 // Add control to item before fetching sprite sheet
663 item->controls.add(control);
664
665 // Fetch the switch sprite sheet if one is specified
666 if (control.image.isNotEmpty())
667 {
668 fetchSwitchSpriteSheet(item, item->controls.size() - 1);
669 }
670 else
671 {
672 }
673 break;
674 }
675
676 case GearControl::Type::Fader:
677 {
678 // Get fader-specific properties
679 control.orientation = controlVar.getProperty("orientation", "vertical").toString();
680 control.length = controlVar.getProperty("length", 100);
681 control.value = controlVar.getProperty("value", 0.0f);
682 control.initialValue = control.value; // Store initial value
683 control.image = controlVar.getProperty("image", "").toString();
684
685 // Add control to item before fetching image
686 item->controls.add(control);
687
688 // Fetch the fader image if one is specified
689 if (control.image.isNotEmpty())
690 {
691 fetchFaderImage(item, item->controls.size() - 1);
692 }
693 else
694 {
695 }
696 break;
697 }
698
699 case GearControl::Type::Knob:
700 {
701 // Get knob-specific properties
702 float startAngle = controlVar.getProperty("startAngle", 0.0f);
703 float endAngle = controlVar.getProperty("endAngle", 360.0f);
704 float value = controlVar.getProperty("value", 0.0f);
705
706 control.startAngle = startAngle;
707 control.endAngle = endAngle;
708 control.value = value;
709 control.initialValue = value; // Store the initial value from schema
710 control.image = controlVar.getProperty("image", "").toString();
711
712 // Handle stepped knobs
713 if (controlVar.hasProperty("steps") && controlVar["steps"].isArray())
714 {
715 juce::Array<juce::var> *stepsArray = controlVar["steps"].getArray();
716 for (const auto &step : *stepsArray)
717 control.steps.add(step);
718 control.currentStepIndex = controlVar.getProperty("currentStepIndex", 0);
719 }
720
721 // Add control to item before fetching image
722 item->controls.add(control);
723
724 // Fetch the knob image if one is specified
725 if (control.image.isNotEmpty())
726 {
727 // Pass the index of the control we just added
728 fetchKnobImage(item, item->controls.size() - 1);
729 }
730 else
731 {
732 }
733 break;
734 }
735
736 case GearControl::Type::Button:
737 {
738 // Get button-specific properties
739 control.momentary = controlVar.getProperty("momentary", false);
740 control.value = controlVar.getProperty("value", 0.0f);
741 control.initialValue = control.value; // Store initial value
742 control.image = controlVar.getProperty("image", "").toString();
743
744 // Parse options array for button states
745 if (controlVar.hasProperty("options") && controlVar["options"].isArray())
746 {
747 auto optionsArray = controlVar["options"].getArray();
748 for (const auto &option : *optionsArray)
749 {
750 if (option.isObject())
751 {
752 GearControl::SwitchOptionFrame frame;
753 frame.value = option.getProperty("value", "").toString();
754 frame.label = option.getProperty("label", "").toString();
755
756 // Parse frame data
757 if (option.hasProperty("frame") && option["frame"].isObject())
758 {
759 auto frameObj = option["frame"];
760 frame.x = frameObj.getProperty("x", 0);
761 frame.y = frameObj.getProperty("y", 0);
762 frame.width = frameObj.getProperty("width", 0);
763 frame.height = frameObj.getProperty("height", 0);
764 }
765
766 control.buttonFrames.add(frame);
767 control.options.add(frame.value);
768 }
769 }
770 }
771
772 // Set currentIndex based on value
773 if (control.momentary)
774 {
775 control.currentIndex = control.value > 0.5f ? 1 : 0;
776 }
777 else
778 {
779 control.currentIndex = (int)control.value;
780 }
781
782 // Add control to item before fetching sprite sheet
783 item->controls.add(control);
784
785 // Fetch the button sprite sheet if one is specified
786 if (control.image.isNotEmpty())
787 {
788 fetchButtonSpriteSheet(item, item->controls.size() - 1);
789 }
790 else
791 {
792 }
793 break;
794 }
795
796 default:
797 // For non-knob controls, just add them
798 item->controls.add(control);
799 break;
800 }
801 }
802 }

804 // Notify that the schema has been loaded
805}

841 struct FaceplateImageDownloader : public juce::Thread

874 // Try to download using the simple API - this is for older JUCE versions
875 std::unique_ptr<juce::InputStream> inputStream = url.createInputStream(false);
876
877 if (inputStream == nullptr || threadShouldExit())
878 {

880 juce::MessageManager::callAsync([this]()
881 {
882 // Clear any existing images
883 if (item != nullptr)

885 item->faceplateImage = juce::Image();

894

896 format = new juce::JPEGImageFormat();
897 else if (urlStr.contains(".png"))
898 format = new juce::PNGImageFormat();
899 else if (urlStr.contains(".gif"))

910 }

914 downloadedImage = juce::ImageFileFormat::loadFrom(*inputStream);
915 }
916

918 {
919 // Need to get back on the message thread to update the UI
920 juce::MessageManager::callAsync([this, downloadedImage]()
921 {
922 // Clear any existing images first
923 if (item != nullptr)
924 {
925 item->faceplateImage = juce::Image();

938 {

940 if (slot && slot->getGearItem() == item)
941 {
942 slot->repaint();
943 }
944 }
945

947 rack->resized();
948 }
949
950 delete this; });
951 }
952 else
953 {
954 // Image loading failed, clean up
955 juce::MessageManager::callAsync([this]()
956 {
957 // Clear any existing images
958 if (item != nullptr)
959 {
960 item->faceplateImage = juce::Image();
961 }
962
963 // Create a placeholder image instead
964 juce::Image placeholderImage(juce::Image::RGB, 200, 100, true);
965 juce::Graphics g(placeholderImage);
966 g.fillAll(juce::Colours::darkgrey);
967 g.setColour(juce::Colours::white);
968 g.drawText("Faceplate Unavailable", placeholderImage.getBounds(), juce::Justification::centred, true);
969
970 // Set as faceplate image
971 if (item != nullptr)
972 {
973 item->faceplateImage = placeholderImage;
974 }
975
976 // Repaint any slots with this item
