95int Rack::getSlotHeight(int slotIndex) const
96{
97 if (slotIndex < 0 || slotIndex >= slots.size())
98 return getDefaultSlotHeight();
99
100 RackSlot *slot = slots[slotIndex];
101 if (slot == nullptr || slot->isAvailable())
102 return getDefaultSlotHeight();
103
104 // If the slot has a gear item with a faceplate image, use the image's height plus padding
105 GearItem *item = slot->getGearItem();
106 if (item != nullptr && item->faceplateImage.isValid())
107 {
108 // Calculate a reasonable height based on the faceplate image
109 // Use aspect ratio of the image, but constrained to reasonable bounds
110 int imageHeight = item->faceplateImage.getHeight();
111 int imageWidth = item->faceplateImage.getWidth();
112
113 if (imageHeight > 0 && imageWidth > 0)
114 {
115 // Calculate what the height would be if the width matched the slot width
116 // Add extra padding for controls and slot UI elements
117 int effectiveSlotWidth = rackContainer->getWidth() - (2 * slotSpacing);
118 int scaledHeight = (imageHeight * effectiveSlotWidth) / imageWidth;
119
120 // Add padding for slot UI elements (buttons, labels, etc.)
121 int paddedHeight = scaledHeight + 40; // 20px padding top and bottom
122
123 // Constrain to reasonable bounds
124 return juce::jlimit(100, 400, paddedHeight);
125 }
126 }
127
128 // Default height if no special considerations apply
129 return getDefaultSlotHeight();

138void Rack::resized()
139{
140 // Size the viewport to fill our area
141 auto area = getLocalBounds();
142 rackViewport->setBounds(area);
143
144 // Get the available width from the viewport
145 const int availableWidth = rackViewport->getWidth();
146
147 // Calculate total height needed for all slots with their variable heights
148 int totalHeight = slotSpacing; // Start with top spacing
149 for (int i = 0; i < slots.size(); ++i)
150 {
151 totalHeight += getSlotHeight(i) + slotSpacing;
152 }
153
154 // Size the container to fit all slots with spacing
155 const int containerWidth = availableWidth; // Use full viewport width
156 rackContainer->setSize(containerWidth, totalHeight);
157
158 // Calculate the slot width based on container width minus margins
159 const int effectiveSlotWidth = containerWidth - (2 * slotSpacing);
160
161 // Position the slots within the container
162 int currentY = slotSpacing;
163 for (int i = 0; i < slots.size(); ++i)
164 {
165 auto *slot = slots[i];
166 int slotHeight = getSlotHeight(i);
167
168 slot->setBounds(
169 slotSpacing,
170 currentY,
171 effectiveSlotWidth,
172 slotHeight);
173
174 currentY += slotHeight + slotSpacing;
175 }

409RackSlot *Rack::findNearestSlot(const juce::Point<int> &position)
410{
411 // This method is kept for compatibility with drops from GearLibrary
412 // but we've simplified it since we're not using it for reordering
413
414 // Convert global position to container coordinates
415 auto containerPos = rackContainer->getLocalPoint(this, position);
416
417 // First check for a direct hit
418 for (auto *slot : slots)
419 {
420 if (slot->getBounds().contains(containerPos))
421 {
422 return slot;
423 }
424 }
425
426 // If no direct hit, find the closest slot
427 RackSlot *bestSlot = nullptr;
428 int bestDistance = std::numeric_limits<int>::max();
429
430 for (auto *slot : slots)
431 {
432 juce::Rectangle<int> slotBounds = slot->getBounds();
433 int distance = slotBounds.getCentre().getDistanceFrom(containerPos);
434
435 if (distance < bestDistance)
436 {
437 bestDistance = distance;
438 bestSlot = slot;
439 }
440 }
441
442 return bestSlot;

571void Rack::parseSchema(const juce::String &schemaData, GearItem *item, std::function<void()> onComplete)
572{
573 // Parse the JSON schema
574 auto schemaJson = juce::JSON::parse(schemaData);
575 if (!schemaJson.isObject())
576 {
577 if (onComplete)
578 onComplete();
579 return;
580 }
581
582 // Look for faceplate image properties
583 juce::StringArray faceplateProperties = {"faceplateImage", "thumbnailImage"};
584 juce::String faceplateImagePath;
585 bool foundFaceplate = false;
586
587 for (const auto &propName : faceplateProperties)
588 {
589 if (schemaJson.hasProperty(propName))
590 {
591 faceplateImagePath = schemaJson.getProperty(propName, "").toString();
592 if (faceplateImagePath.isNotEmpty())
593 {
594 foundFaceplate = true;
595
596 // Store the path in the gear item
597 item->faceplateImagePath = faceplateImagePath;
598
599 // Fetch the faceplate image
600 fetchFaceplateImage(item);
601 break;
602 }
603 }
604 }
605
606 if (!foundFaceplate)
607 {
608 }
609
610 // Update controls if available
611 if (schemaJson.hasProperty("controls") && schemaJson["controls"].isArray())
612 {
613 item->controls.clear();
614
615 auto controlsArray = schemaJson["controls"].getArray();
616 for (auto &controlVar : *controlsArray)
617 {
618 if (!controlVar.isObject())
619 continue;
620
621 // Get control type
622 GearControl::Type controlType = GearControl::Type::Button;
623 juce::String controlTypeStr = controlVar.getProperty("type", "button").toString().toLowerCase();
624 if (controlTypeStr == "fader")
625 controlType = GearControl::Type::Fader;
626 else if (controlTypeStr == "switch")
627 controlType = GearControl::Type::Switch;
628 else if (controlTypeStr == "knob")
629 controlType = GearControl::Type::Knob;
630
631 // Get control name and ID
632 juce::String controlName = controlVar.getProperty("label", "");
633 juce::String controlId = controlVar.getProperty("id", ""); // Get ID from schema
634
635 // If no ID is provided, generate one from the label
636 if (controlId.isEmpty())
637 {
638 controlId = controlName.toLowerCase().replaceCharacters(" ", "-");
639 }
640
641 // Check if we already have a control with this ID
642 bool controlExists = false;
643 for (const auto &existingControl : item->controls)
644 {
645 if (existingControl.id == controlId)
646 {
647 controlExists = true;
648 break;
649 }
650 }
651
652 if (controlExists)
653 continue;
654
655 // Get position
656 juce::Rectangle<float> position;
657 if (controlVar.hasProperty("position") && controlVar["position"].isObject())
658 {
659 position.setX(controlVar["position"].getProperty("x", 0.0f));
660 position.setY(controlVar["position"].getProperty("y", 0.0f));
661 position.setWidth(controlVar["position"].getProperty("width", 0.0f));
662 position.setHeight(controlVar["position"].getProperty("height", 0.0f));
663 }
664
665 // Create control
666 GearControl control(controlType, controlName, position);
667 control.id = controlId; // Set the control ID
668 control.value = controlVar.getProperty("value", 0.0f);
669
670 // Set type-specific properties
671 switch (control.type)
672 {
673 case GearControl::Type::Switch:
674 {
675 // Get switch-specific properties
676 control.orientation = controlVar.getProperty("orientation", "vertical").toString();
677 control.currentIndex = controlVar.getProperty("currentIndex", 0);
678 control.value = (float)control.currentIndex; // Set value based on currentIndex
679 control.initialValue = control.value; // Store initial value
680 control.image = controlVar.getProperty("image", "").toString();
681
682 // Parse options array
683 if (controlVar.hasProperty("options") && controlVar["options"].isArray())
684 {
685 auto optionsArray = controlVar["options"].getArray();
686 for (const auto &option : *optionsArray)
687 {
688 if (option.isObject())
689 {
690 GearControl::SwitchOptionFrame frame;
691 frame.value = option.getProperty("value", "").toString();
692 frame.label = option.getProperty("label", "").toString();
693
694 // Parse frame data
695 if (option.hasProperty("frame") && option["frame"].isObject())
696 {
697 auto frameObj = option["frame"];
698 frame.x = frameObj.getProperty("x", 0);
699 frame.y = frameObj.getProperty("y", 0);
700 frame.width = frameObj.getProperty("width", 0);
701 frame.height = frameObj.getProperty("height", 0);
702 }
703
704 control.switchFrames.add(frame);
705 control.options.add(frame.value);
706 }
707 }
708 }
709
710 // Add control to item before fetching sprite sheet
711 item->controls.add(control);
712
713 // Fetch the switch sprite sheet if one is specified
714 if (control.image.isNotEmpty())
715 {
716 fetchSwitchSpriteSheet(item, item->controls.size() - 1);
717 }
718 else
719 {
720 }
721 break;
722 }
723
724 case GearControl::Type::Fader:
725 {
726 // Get fader-specific properties
727 control.orientation = controlVar.getProperty("orientation", "vertical").toString();
728 control.length = controlVar.getProperty("length", 100);
729 control.value = controlVar.getProperty("value", 0.0f);
730 control.initialValue = control.value; // Store initial value
731 control.image = controlVar.getProperty("image", "").toString();
732
733 // Add control to item before fetching image
734 item->controls.add(control);
735
736 // Fetch the fader image if one is specified
737 if (control.image.isNotEmpty())
738 {
739 fetchFaderImage(item, item->controls.size() - 1);
740 }
741 else
742 {
743 }
744 break;
745 }
746
747 case GearControl::Type::Knob:
748 {
749 // Get knob-specific properties
750 float startAngle = controlVar.getProperty("startAngle", 0.0f);
751 float endAngle = controlVar.getProperty("endAngle", 360.0f);
752 float value = controlVar.getProperty("value", 0.0f);
753
754 control.startAngle = startAngle;
755 control.endAngle = endAngle;
756 control.value = value;
757 control.initialValue = value; // Store the initial value from schema
758 control.image = controlVar.getProperty("image", "").toString();
759
760 // Handle stepped knobs
761 if (controlVar.hasProperty("steps") && controlVar["steps"].isArray())
762 {
763 juce::Array<juce::var> *stepsArray = controlVar["steps"].getArray();
764 for (const auto &step : *stepsArray)
765 control.steps.add(step);
766 control.currentStepIndex = controlVar.getProperty("currentStepIndex", 0);
767 }
768
769 // Add control to item before fetching image
770 item->controls.add(control);
771
772 // Fetch the knob image if one is specified
773 if (control.image.isNotEmpty())
774 {
775 // Pass the index of the control we just added
776 fetchKnobImage(item, item->controls.size() - 1);
777 }
778 else
779 {
780 }
781 break;
782 }
783
784 case GearControl::Type::Button:
785 {
786 // Get button-specific properties
787 control.momentary = controlVar.getProperty("momentary", false);
788 control.value = controlVar.getProperty("value", 0.0f);
789 control.initialValue = control.value; // Store initial value
790 control.image = controlVar.getProperty("image", "").toString();
791
792 // Parse options array for button states
793 if (controlVar.hasProperty("options") && controlVar["options"].isArray())
794 {
795 auto optionsArray = controlVar["options"].getArray();
796 for (const auto &option : *optionsArray)
797 {
798 if (option.isObject())
799 {
800 GearControl::SwitchOptionFrame frame;
801 frame.value = option.getProperty("value", "").toString();
802 frame.label = option.getProperty("label", "").toString();
803
804 // Parse frame data
805 if (option.hasProperty("frame") && option["frame"].isObject())
806 {
807 auto frameObj = option["frame"];
808 frame.x = frameObj.getProperty("x", 0);
809 frame.y = frameObj.getProperty("y", 0);
810 frame.width = frameObj.getProperty("width", 0);
811 frame.height = frameObj.getProperty("height", 0);
812 }
813
814 control.buttonFrames.add(frame);
815 control.options.add(frame.value);
816 }
817 }
818 }
819
820 // Set currentIndex based on value
821 if (control.momentary)
822 {
823 control.currentIndex = control.value > 0.5f ? 1 : 0;
824 }
825 else
826 {
827 control.currentIndex = (int)control.value;
828 }
829
830 // Add control to item before fetching sprite sheet
831 item->controls.add(control);
832
833 // Fetch the button sprite sheet if one is specified
834 if (control.image.isNotEmpty())
835 {
836 fetchButtonSpriteSheet(item, item->controls.size() - 1);
837 }
838 else
839 {
840 }
841 break;
842 }
843
844 default:
845 // For non-knob controls, just add them
846 item->controls.add(control);
847 break;
848 }
849 }
850 }
851
852 // Call the completion callback if provided
853 if (onComplete)
854 onComplete();
855}
