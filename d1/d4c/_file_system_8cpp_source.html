199juce::String FileSystem::getFileName(const juce::String &path)
200{
201 if (path.isEmpty())
202 {
203 return {};
204 }
205
206 // Handle relative paths by extracting just the filename part
207 if (!juce::File::isAbsolutePath(path))
208 {
209 // For relative paths, just return the last component
210 int lastSlash = path.lastIndexOfChar('/');
211 if (lastSlash >= 0)
212 {
213 juce::String result = path.substring(lastSlash + 1);
214 return result;
215 }
216 else
217 {
218 // No slashes, return the whole path as filename
219 return path;
220 }
221 }
222
223 // For absolute paths, use JUCE File
224 juce::File file(path);
225 juce::String result = file.getFileName();
226 return result;
227}
