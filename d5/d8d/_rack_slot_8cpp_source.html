471void RackSlot::drawFader(juce::Graphics &g, const GearControl &control, int x, int y)
472{
473 const bool isVertical = control.orientation == "vertical";
474 const float faderLength = control.length * currentFaceplateScale;
475
476 // Calculate base dimensions from the image if available, otherwise use defaults
477 float baseWidth, baseHeight;
478 if (control.faderImage.isValid())
479 {
480 baseWidth = (float)control.faderImage.getWidth();
481 baseHeight = (float)control.faderImage.getHeight();
482 }
483 else
484 {
485 baseWidth = 20.0f;
486 baseHeight = 20.0f;
487 }
488
489 const float faderWidth = baseWidth * currentFaceplateScale;
490 const float handleSize = std::max(baseWidth, baseHeight) * currentFaceplateScale;
491
492 // Calculate the fader bounds based on orientation
493 juce::Rectangle<float> faderBounds;
494 if (isVertical)
495 {
496 faderBounds = juce::Rectangle<float>(x, y, faderWidth, faderLength);
497 }
498 else
499 {
500 faderBounds = juce::Rectangle<float>(x, y, faderLength, faderWidth);
501 }
502
503 // Calculate the handle position based on the control value
504 float handleX, handleY;
505 if (isVertical)
506 {
507 handleX = x;
508 handleY = y + (1.0f - control.value) * faderLength;
509 }
510 else
511 {
512 handleX = x + control.value * faderLength;
513 handleY = y;
514 }
515
516 // Draw the fader image at the handle position
517 if (control.faderImage.isValid())
518 {
519 // Scale the handle size based on the image's aspect ratio
520 float imageWidth = (float)control.faderImage.getWidth();
521 float imageHeight = (float)control.faderImage.getHeight();
522 float aspectRatio = imageWidth / imageHeight;
523
524 float scaledWidth, scaledHeight;
525 if (isVertical)
526 {
527 scaledHeight = handleSize;
528 scaledWidth = handleSize * aspectRatio;
529 }
530 else
531 {
532 scaledWidth = handleSize;
533 scaledHeight = handleSize / aspectRatio;
534 }
535
536 // Draw the fader image centered at the handle position
537 g.drawImageWithin(control.faderImage,
538 handleX - scaledWidth / 2,
539 handleY - scaledHeight / 2,
540 scaledWidth,
541 scaledHeight,
542 juce::RectanglePlacement::centred);
543 }
544 else
545 {
546 // Fallback to basic handle drawing if no image is provided
547 g.setColour(juce::Colours::white);
548 if (isVertical)
549 {
550 g.fillRoundedRectangle(x - handleSize / 2, handleY - handleSize / 2,
551 faderWidth + handleSize, handleSize, 4.0f);
552 }
553 else
554 {
555 g.fillRoundedRectangle(handleX - handleSize / 2, y - handleSize / 2,
556 handleSize, faderWidth + handleSize, 4.0f);
557 }
558 }

569void RackSlot::drawKnob(juce::Graphics &g, const GearControl &control, int x, int y)
570{
571 // Calculate knob size based on faceplate scale factor
572 float knobSize;
573 if (control.loadedImage.isValid())
574 {
575 // Use the original image dimensions as the base size
576 float originalWidth = (float)control.loadedImage.getWidth();
577 float originalHeight = (float)control.loadedImage.getHeight();
578 // Use the larger dimension to ensure the knob is properly sized
579 knobSize = std::max(originalWidth, originalHeight) * currentFaceplateScale;
580 }
581 else
582 {
583 // Fallback to standard size if no image
584 const float baseKnobSize = 40.0f;
585 knobSize = baseKnobSize * currentFaceplateScale;
586 }
587
588 // Create knob bounds using the transformed coordinates and scaled size
589 juce::Rectangle<float> knobBounds(x, y, knobSize, knobSize);
590
591 // Draw the knob image if available
592 if (control.loadedImage.isValid())
593 {
594 // Save the current graphics state
595 g.saveState();
596
597 // Use the control value directly as degrees, but subtract 180 to align with JUCE's coordinate system
598 // where 0 is at 12 o'clock and we want 0 to be at 6 o'clock
599 float angle = control.value - 180.0f;
600
601 // Translate to the center of the knob
602 g.addTransform(juce::AffineTransform::translation(knobBounds.getCentreX(), knobBounds.getCentreY()));
603
604 // Rotate around the center
605 g.addTransform(juce::AffineTransform::rotation(juce::degreesToRadians(angle)));
606
607 // Translate back and draw the image centered
608 g.addTransform(juce::AffineTransform::translation(-knobBounds.getCentreX(), -knobBounds.getCentreY()));
609
610 // Draw the image
611 g.drawImageWithin(control.loadedImage,
612 knobBounds.getX(), knobBounds.getY(),
613 knobBounds.getWidth(), knobBounds.getHeight(),
614 juce::RectanglePlacement::centred);
615
616 // Restore the graphics state
617 g.restoreState();
618 }
619 else
620 {
621 // Fallback to basic drawing if no image is provided
622 g.setColour(juce::Colours::darkgrey);
623 g.fillEllipse(knobBounds);
624 g.setColour(juce::Colours::black);
625 g.drawEllipse(knobBounds, 1.0f);
626
627 // Draw position indicator
628 g.setColour(juce::Colours::white);
629 float angle = control.value - 180.0f; // Subtract 90 to align with JUCE's coordinate system
630 float radius = knobBounds.getWidth() * 0.4f;
631 float centreX = knobBounds.getCentreX();
632 float centreY = knobBounds.getCentreY();
633
634 // Convert angle to radians
635 float angleRad = angle * (juce::MathConstants<float>::pi / 180.0f);
636 float endX = centreX + radius * std::cos(angleRad);
637 float endY = centreY + radius * std::sin(angleRad);
638
639 g.drawLine(centreX, centreY, endX, endY, 2.0f);
640 }

691void RackSlot::updateButtonStates()
692{
693 // Disable buttons if there's no gear item
694 bool hasGear = !isAvailable() && gearItem != nullptr;
695
696 // Make sure buttons exist before using them
697 if (upButton == nullptr || downButton == nullptr || removeButton == nullptr)
698 return;
699
700 // Up button should be disabled for the first slot
701 upButton->setEnabled(hasGear && index > 0);
702
703 // Down button should be disabled for the last slot
704 // We need to check the total number of slots from the parent Rack
705 Rack *parentRack = dynamic_cast<Rack *>(findParentRackComponent());
706 if (parentRack != nullptr)
707 {
708 int totalSlots = parentRack->getNumSlots();
709 downButton->setEnabled(hasGear && index < totalSlots - 1);
710 }
711 else
712 {
713 downButton->setEnabled(hasGear); // Fallback if we can't get the rack
714 }
715
716 // Remove button is only enabled when there's a gear item
717 removeButton->setEnabled(hasGear);

890void RackSlot::mouseDrag(const juce::MouseEvent &e)
891{
892 if (!isDragging || activeControl == nullptr)
893 {
894 return;
895 }
896
897 // Calculate faceplate area
898 juce::Rectangle<int> faceplateArea = getLocalBounds().reduced(10);
899 faceplateArea.removeFromTop(20); // Remove space for name
900
901 // Calculate control bounds
902 int x = faceplateArea.getX() + (int)(activeControl->position.getX() * faceplateArea.getWidth());
903 int y = faceplateArea.getY() + (int)(activeControl->position.getY() * faceplateArea.getHeight());
904
905 switch (activeControl->type)
906 {
907 case GearControl::Type::Switch:
908 {
909 const bool isVertical = activeControl->orientation == "vertical";
910 const int numOptions = activeControl->options.size();
911 const float faderLength = activeControl->length * currentFaceplateScale;
912
913 // Calculate the drag distance along the orientation axis
914 float dragDistance;
915 if (isVertical)
916 {
917 dragDistance = e.position.y - dragStartPos.y;
918 // For vertical switches, invert the drag direction to match natural movement
919 dragDistance = -dragDistance;
920 }
921 else
922 {
923 dragDistance = e.position.x - dragStartPos.x;
924 }
925
926 // Calculate the total range of movement using fader length
927 float totalRange = faderLength;
928 float optionSize = totalRange / numOptions;
929
930 // Calculate the new index based on drag distance
931 float newIndex = dragStartValue + (dragDistance / optionSize);
932
933 // Clamp the index to valid range and round to nearest option
934 newIndex = juce::jlimit(0.0f, (float)(numOptions - 1), newIndex);
935 int newIndexInt = juce::roundToInt(newIndex);
936
937 // Update the control
938 if (newIndexInt != activeControl->currentIndex)
939 {
940 activeControl->currentIndex = newIndexInt;
941 activeControl->value = (float)newIndexInt;
942 repaint();
943 // Find the control index and notify
944 if (gearItem != nullptr)
945 {
946 for (int i = 0; i < gearItem->controls.size(); ++i)
947 {
948 if (&gearItem->controls.getReference(i) == activeControl)
949 {
950 notifyRackOfControlChanged(i);
951 break;
952 }
953 }
954 }
955 }
956 break;
957 }
958 case GearControl::Type::Fader:
959 {
960 const bool isVertical = activeControl->orientation == "vertical";
961 const float faderLength = activeControl->length * currentFaceplateScale;
962 float newValue;
963
964 if (isVertical)
965 {
966 // For vertical faders, use Y position relative to fader track
967 float trackY = y;
968 float trackHeight = faderLength;
969 float normalizedY = 1.0f - (float)(e.position.y - trackY) / trackHeight;
970 newValue = juce::jlimit(0.0f, 1.0f, normalizedY);
971 }
972 else
973 {
974 // For horizontal faders, use X position relative to fader track
975 float trackX = x;
976 float trackWidth = faderLength;
977 float normalizedX = (float)(e.position.x - trackX) / trackWidth;
978 newValue = juce::jlimit(0.0f, 1.0f, normalizedX);
979 }
980
981 activeControl->value = newValue;
982 repaint();
983 // Find the control index and notify
984 if (gearItem != nullptr)
985 {
986 for (int i = 0; i < gearItem->controls.size(); ++i)
987 {
988 if (&gearItem->controls.getReference(i) == activeControl)
989 {
990 notifyRackOfControlChanged(i);
991 break;
992 }
993 }
994 }
995 break;
996 }
997 case GearControl::Type::Knob:
998 {
999 // Calculate vertical movement since drag start
1000 float deltaY = dragStartPos.y - e.position.y;
1001
1002 // Scale the movement to control sensitivity
1003 const float sensitivity = 0.5f; // Adjust this value to change knob sensitivity
1004 float deltaAngle = deltaY * sensitivity;
1005
1006 // Update the control value (angle in degrees)
1007 float newValue = dragStartValue + deltaAngle;
1008
1009 // Clamp the value between startAngle and endAngle
1010 newValue = juce::jlimit(activeControl->startAngle, activeControl->endAngle, newValue);
1011
1012 // If this is a stepped knob, snap to the nearest step
1013 if (!activeControl->steps.isEmpty())
1014 {
1015 float closestStep = activeControl->steps[0];
1016 float minDistance = std::abs(newValue - closestStep);
1017
1018 // Find the closest step angle
1019 for (float step : activeControl->steps)
1020 {
1021 float distance = std::abs(newValue - step);
1022 if (distance < minDistance)
1023 {
1024 minDistance = distance;
1025 closestStep = step;
1026 }
1027 }
1028
1029 newValue = closestStep;
1030 }
1031
1032 activeControl->value = newValue;
1033
1034 repaint();
1035 // Find the control index and notify
1036 if (gearItem != nullptr)
1037 {
1038 for (int i = 0; i < gearItem->controls.size(); ++i)
1039 {
1040 if (&gearItem->controls.getReference(i) == activeControl)
1041 {
1042 notifyRackOfControlChanged(i);
1043 break;
1044 }
1045 }
1046 }
1047 break;
1048 }
1049 case GearControl::Type::Button:
1050 handleButtonInteraction(*activeControl);
1051 repaint();
1052 // Find the control index and notify
1053 if (gearItem != nullptr)
1054 {
1055 for (int i = 0; i < gearItem->controls.size(); ++i)
1056 {
1057 if (&gearItem->controls.getReference(i) == activeControl)
1058 {
1059 notifyRackOfControlChanged(i);
1060 break;
1061 }
1062 }
1063 }
1064 break;
1065 default:
1066 break; // Other controls don't need drag handling
1067 }
1068}

1135GearControl *RackSlot::findControlAtPosition(const juce::Point<float> &position, const juce::Rectangle<int> &faceplateArea)
1136{
1137 if (gearItem == nullptr)
1138 return nullptr;
1139
1140 for (auto &control : gearItem->controls)
1141 {
1142 // Calculate control bounds
1143 int x = faceplateArea.getX() + (int)(control.position.getX() * faceplateArea.getWidth());
1144 int y = faceplateArea.getY() + (int)(control.position.getY() * faceplateArea.getHeight());
1145
1146 // Calculate actual rendered bounds based on control type
1147 juce::Rectangle<float> controlBounds;
1148
1149 switch (control.type)
1150 {
1151 case GearControl::Type::Switch:
1152 {
1153 if (control.switchSpriteSheet.isValid() && control.switchFrames.size() > 0)
1154 {
1155 // Use sprite frame dimensions (same as drawing)
1156 int currentIndex = control.currentIndex;
1157 if (currentIndex >= control.switchFrames.size())
1158 currentIndex = 0;
1159
1160 const auto &frame = control.switchFrames[currentIndex];
1161 float scaledFrameWidth = frame.width * currentFaceplateScale;
1162 float scaledFrameHeight = frame.height * currentFaceplateScale;
1163 controlBounds = juce::Rectangle<float>(x, y, scaledFrameWidth, scaledFrameHeight);
1164 }
1165 else
1166 {
1167 // Fallback to basic dimensions (same as drawing)
1168 const int switchWidth = 30;
1169 const int switchHeight = 60;
1170 controlBounds = juce::Rectangle<float>(x, y, switchWidth, switchHeight);
1171 }
1172 break;
1173 }
1174
1175 case GearControl::Type::Button:
1176 {
1177 if (control.buttonSpriteSheet.isValid() && control.buttonFrames.size() > 0)
1178 {
1179 // Use sprite frame dimensions (same as drawing)
1180 int frameIndex = control.currentIndex;
1181 if (frameIndex >= control.buttonFrames.size())
1182 frameIndex = 0;
1183
1184 const auto &frame = control.buttonFrames[frameIndex];
1185 float scaledFrameWidth = frame.width * currentFaceplateScale;
1186 float scaledFrameHeight = frame.height * currentFaceplateScale;
1187 controlBounds = juce::Rectangle<float>(x, y, scaledFrameWidth, scaledFrameHeight);
1188 }
1189 else
1190 {
1191 // Fallback to basic dimensions (same as drawing)
1192 const float buttonWidth = 30.0f * currentFaceplateScale;
1193 const float buttonHeight = 30.0f * currentFaceplateScale;
1194 controlBounds = juce::Rectangle<float>(x, y, buttonWidth, buttonHeight);
1195 }
1196 break;
1197 }
1198
1199 case GearControl::Type::Fader:
1200 {
1201 // Use the same handle calculation logic as drawing
1202 const bool isVertical = control.orientation == "vertical";
1203 const float faderLength = control.length * currentFaceplateScale;
1204 const float faderWidth = 20.0f * currentFaceplateScale;
1205 const float handleSize = 10.0f * currentFaceplateScale;
1206
1207 // Calculate handle position (same as drawing)
1208 float handleX, handleY;
1209 if (isVertical)
1210 {
1211 handleX = x;
1212 handleY = y + (1.0f - control.value) * faderLength;
1213 }
1214 else
1215 {
1216 handleX = x + control.value * faderLength;
1217 handleY = y;
1218 }
1219
1220 // Create handle bounds (same as drawing)
1221 controlBounds = juce::Rectangle<float>(
1222 handleX - handleSize / 2,
1223 handleY - handleSize / 2,
1224 isVertical ? faderWidth + handleSize : handleSize,
1225 isVertical ? handleSize : faderWidth + handleSize);
1226 break;
1227 }
1228
1229 case GearControl::Type::Knob:
1230 {
1231 // Use the same logic as drawKnob() to calculate knob size
1232 float knobSize;
1233 if (control.loadedImage.isValid())
1234 {
1235 // Use the original image dimensions as the base size (same as drawing)
1236 float originalWidth = (float)control.loadedImage.getWidth();
1237 float originalHeight = (float)control.loadedImage.getHeight();
1238 // Use the larger dimension to ensure the knob is properly sized
1239 knobSize = std::max(originalWidth, originalHeight) * currentFaceplateScale;
1240 }
1241 else
1242 {
1243 // Fallback to standard size if no image (same as drawing)
1244 const float baseKnobSize = 40.0f;
1245 knobSize = baseKnobSize * currentFaceplateScale;
1246 }
1247 controlBounds = juce::Rectangle<float>(x, y, knobSize, knobSize);
1248 break;
1249 }
1250
1251 default:
1252 controlBounds = juce::Rectangle<float>(x, y, 40, 40);
1253 }
1254
1255 // Check if position is within the actual rendered bounds
1256 if (controlBounds.contains(position))
1257 return &control;
1258 }
1259
1260 return nullptr;
1261}
